export class CmDin
 extends HTMLElement {
 connectedCallback() {
  setTimeout(
   () => this.prepara(),
   100)
 }
 prepara() {
  /**
   * @private
   * @type {HTMLInputElement|
   *  HTMLSelectElement|
   *  HTMLOutputElement|
   *  null} */
  this._elem = this.
   querySelector(
    "input, select, output")
  if (this._elem != null) {
   this.addEventListener("input",
    () => this.analiza())
  }
  this.analiza();
 }
 analiza() {
  const e = this._elem;
  if (e != null) {
   if (!e.validity.badInput &&
    e.value === "") {
    e.classList.remove("valor")
   } else {
    e.classList.add("valor")
   }
  }
 }
}
customElements.
 define("cm-din", CmDin)
