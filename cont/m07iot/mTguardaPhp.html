<?php
require_once
 "./modelo/Dispositivo.php";
require_once
 "./dao/dispositivoModifica.php";
require_once
 "./dao/historicoAgrega.php";

function guarda($valor)
{
 try {
  $dispositivo = new Dispositivo();
  $dispositivo->id = "1";
  $dispositivo->valor = $valor;
  dispositivoModifica(
   $dispositivo
  );
  historicoAgrega($dispositivo);
 } catch (\Throwable $e) {
  echo $e->getMessage();
 }
}
