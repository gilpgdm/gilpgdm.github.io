<?php
require_once __DIR__ .
 "/../modelo/Dispositivo.php";
 require_once __DIR__ .
 "/AccesoBd.php";

function dispositivoAgrega(
 Dispositivo $dispositivo
) {
 $dispositivo->valida();
 $con = AccesoBd::getCon();
 $stmt = $con->prepare(
  "INSERT INTO DISPOSITIVO
    (DIS_ID, DIS_VALOR)
   VALUES
    (:id, :valor)"
 );
 $stmt->execute([
  ":id" => $dispositivo->id,
  ":valor" => $dispositivo->valor
 ]);
}
