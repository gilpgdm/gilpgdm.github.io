<?php
require_once __DIR__ .
 "/../modelo/Usuario.php";
require_once __DIR__ .
 "/AccesoBd.php";

function usuarioBusca(string $cue)
{
 $con = AccesoBd::getCon();
 $stmt = $con->prepare(
  "SELECT
    USU_ID as id,
    USU_CUE as cue,
    USU_MATCH as match
   FROM USUARIO
   WHERE USU_CUE = :cue"
 );
 $stmt->execute(
  [":cue" => $cue]
 );
 $stmt->setFetchMode(
  PDO::FETCH_CLASS,
  "Usuario"
 );
 return $stmt->fetch();
}
