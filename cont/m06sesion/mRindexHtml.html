<body>
 <h1>Autenticaci贸n</h1>
 <form id="sesion" hidden>
  <output name="cue"></output>
  <button id="terminar"
    type="button">
   Terminar sesi贸n
  </button>
 </form>
 <form id="login" hidden>
  <p>
   <label>
    Cue
    <input name="cue">
   </label>
  </p>
  <p>
   <label>
    Match
    <input type="password"
      name="match">
   </label>
  </p>
  <p>
   <button>Iniciar sesi贸n</button>
  </p>
 </form>
 <script type="module">
  import {
   ejecuta
  } from "./lib/js/ejecuta.js"
  import {
   muestraError
  } from "./lib/js/muestraError.js"
  import {
   submit
  } from "./lib/js/submit.js"

  terminar.addEventListener(
   "click", terminaSesion)
  login.addEventListener("submit",
   iniciaSesion)

  recuperaSesion()

  async function muestraLogin() {
   sesion.hidden = true
   sesion.cue.value = ""
   login.cue.value = ""
   login.match.value = ""
   login.hidden = false
  }

  /** @param {string} usuario */
  async function
   muestraSesion(usuario) {
   sesion.hidden = false
   sesion.cue.value = usuario
   login.match.value = ""
   login.hidden = true
  }

  export async function
   recuperaSesion() {
   try {
    const json = await ejecuta(
     fetch("srv/SrvSesion.php"))
    if (json.cue === "") {
     muestraLogin()
    } else {
     muestraSesion(json.cue)
    }
   } catch (e) {
    muestraError(e)
   }
  }

  /** @param {Event} evt */
  export
   async function iniciaSesion(
    evt) {
   try {
    evt.preventDefault()
    const json = await submit(evt,
     "srv/SrvLogin.php", login)
    muestraSesion(json.cue)
   } catch (e) {
    muestraError(e)
   }
  }

  export
   async function terminaSesion() {
   try {
    await ejecuta(
     fetch("srv/SrvLogout.php"))
    muestraLogin()
   } catch (e) {
    muestraError(e)
   }
  }
 </script>
