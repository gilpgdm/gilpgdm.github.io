import {
 ejecuta
} from "../lib/ejecuta.js";
import {
 muestraError
} from "../lib/muestraError.js";
import {
 submit
} from "../lib/submit.js";

/** @type {HTMLFormElement|null} */
const sesion = document["sesion"]
/** @type {HTMLFormElement|null} */
const login = document["login"]

let esHumano = false

if (sesion !== null) {
 sesion.terminar.addEventListener(
  "click", terminaSesi贸n)
}
if (login !== null) {
 login.addEventListener("submit",
  iniciaSesi贸n)
 login.humano.addEventListener(
  "input", calculaQueSeaHumano)
}

recuperaSesion();

function calculaQueSeaHumano() {
 if (login !== null) {
  esHumano =
   login.humano.valueAsNumber > 0
 }
}

async function recuperaSesion() {
 try {
  if (sesion !== null) {
   const json = await ejecuta(
    fetch("srv/SrvSesion.php"))
   if (json.cue === "") {
    muestraLogin()
   } else {
    muestraSesion(json.cue)
   }
  }
 } catch (e) {
  muestraError(e)
 }
}

/** @param {Event} evt */
async function iniciaSesi贸n(evt) {
 try {
  evt.preventDefault();
  if (!esHumano) {
   throw new Error(
    "Comprueba que eres humano.")
  }
  if (login !== null
   && sesion !== null) {
   const json = await submit(evt,
    "srv/SrvLogin.php", login)
   muestraSesion(json.cue)
  }
 } catch (e) {
  muestraError(e)
 }
}

async function terminaSesi贸n() {
 try {
  await ejecuta(
   fetch("srv/SrvLogout.php"))
  muestraLogin()
 } catch (e) {
  muestraError(e)
 }
}

async function muestraLogin() {
 if (sesion !== null) {
  sesion.hidden = true
  sesion.cue.value = ""
 }
 if (login !== null) {
  login.cue.value = ""
  login.match.value = ""
  login.humano.valueAsNumber = 0
  esHumano = false
  login.hidden = false
 }
}

/**
 * @param {string} usuario
 */
async function
 muestraSesion(usuario) {
 if (sesion !== null) {
  sesion.hidden = false
  sesion.cue.value = usuario
 }
 if (login !== null) {
  login.match.value = ""
  login.hidden = true
 }
}
