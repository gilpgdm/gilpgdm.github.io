• Las barras de aplicación en Material Design están definidas en https://material.io/components/app-bars-top.

  • La distribución básica de las aplicaciónes móviles incluye una barra de aplicación.

  • En iOS el título se muestra centrado.

  /* Custom properties */
  html {
   --colOnBg: #1C1B1F;
   --colBg: #FFFBFE;
   --colOnSrf: #1C1B1F;
   --colSrf: #FFFBFE;

 <script>
  // @ts-check
  const CLASE_ELEVACION2 =
   "elevacion2"

  class CnAppBar
   extends HTMLElement {
   constructor() {
    super()
    /** @type {number} */
    this.posY = 0
    /** @type {boolean} */
    this.procesando = false
   }
   connectedCallback() {
    window.addEventListener(
     "scroll",
     () => this.onScroll())
   }
   onScroll() {
    this.posY = window.scrollY
    if (!this.procesando) {
     window.requestAnimationFrame(
      () => this.avanza())
    }
    this.procesando = true
   }
   avanza() {
    if (this.posY === 0) {
     this.classList.
      remove(CLASE_ELEVACION2)
    } else {
     this.classList.
      add(CLASE_ELEVACION2)
    }
    this.procesando = false
   }
  }
  customElements.
   define("cn-app-bar", CnAppBar)
 </script>

    0 6px 12px var(--colSh);
  }

  cn-app-bar {
   position: sticky;
   top: 0;
   margin: 0;
   display: block;

  }

  }
