38 /**
39 * Cada 20 segundos (2000 milisegundos) después de la última
40 * sincronización, los datos se envían al servidor para volver a
41 * sincronizarse con los datos del servidor.
42 */
43 const MILISEGUNDOS_PARA_VOLVER_A_SINCRONIZAR = 20000
44 let noHaSincronizado = true

46 pasatiempoConsultaNoEliminados()
47 .then(datos => {
48 renderiza(datos)

56 async function renderiza(pasatiempos) {
57
58 let render = ""
59 for (const modelo of pasatiempos) {
60 const nombre = htmlentities(modelo.nombre)
61 const searchParams = new URLSearchParams([["uuid", modelo.uuid]])
62 const params = htmlentities(searchParams.toString())
63 render += /* html */
64 `<li>
65 <p><a href="modifica.html?${params}">${nombre}</a></p>
66 </li>`
67 }
68 lista.innerHTML = render
69 }
70
71 async function sincroniza() {
72 try {
73 if (navigator.onLine) {
74 const todos = await pasatiempoConsultaTodos()
75 const respuesta = await enviaJson("srv/sincroniza.php", todos)
76 const datosNuevos = respuesta.body
77 await pasatiemposReemplaza(datosNuevos)
78 renderiza(datosNuevos)
79 }
80 } catch (error) {
81 muestraError(error)
82 }
83 esperaUnPocoYLanzaSincronizacion()
84 }
85
86 function esperaUnPocoYLanzaSincronizacion() {
87 setTimeout(sincroniza, MILISEGUNDOS_PARA_VOLVER_A_SINCRONIZAR)
88 }
