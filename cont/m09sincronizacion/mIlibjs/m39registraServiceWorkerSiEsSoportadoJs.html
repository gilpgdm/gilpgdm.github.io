import {
 txtRegistrado
} from "./txt/txtRegistrado.js"
import {
 muestraError
} from "./muestraError.js"

/**
 * @param {string | URL} sw URL del
 *  service worker
 */
export async function
 registraServiceWorkerSiEsSoportado(
  sw) {
 try {
  if ("serviceWorker"
   in navigator) {
   const registro =
    await navigator.
     serviceWorker.
     register(sw)
   console.log(sw, txtRegistrado())
   console.log(registro)
  }
 } catch (e) {
  muestraError(e)
 }
}
