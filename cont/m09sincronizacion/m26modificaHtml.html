<!DOCTYPE html>
<html lang="es">

<head>
 <meta charset="UTF-8">
 <meta name="viewport"
   content="width=device-width">
 <title>Modificar</title>
</head>

<body>
 <h1>Modificar</h1>
 <form id="form">
  <p>
   <label>
    Nombre
    <input id="txtNombre" required>
   </label>
  </p>
  <p>
   <a
     href="index.html">Cancelar</a>
   <button>Guardar</button>
   <button id="btnElimina"
     type="button">
    Eliminar
   </button>
  </p>
 </form>
 <script type="module">
  import {
   muestraError
  } from "./lib/js/muestraError.js"
  import "./js/config.js"
  import {
   pasatiempoBusca
  } from "./js/dao/pasatiempoBusca.js"
  import {
   pasatiempoElimina
  } from "./js/dao/pasatiempoElimina.js"
  import {
   pasatiempoModifica
  } from "./js/dao/pasatiempoModifica.js"
  import {
   txtConfirmaEliminacion
  } from "./js/textos/txtConfirmaEliminacion.js"

  const params =
   new URL(location.href).
    searchParams
  const uuid = params.get("uuid")

  form.addEventListener("submit",
   guarda)
  btnElimina.addEventListener(
   "click", elimina)

  busca()

  async function busca() {
   try {
    if (uuid !== null) {
     const modelo =
      await pasatiempoBusca(uuid)
     const nombre =
      typeof modelo.nombre ===
       "string" ?
       modelo.nombre
       : ""
     document.title = nombre
     txtNombre.value = nombre
    }
   } catch (e) {
    muestraError(e)
   }
  }

  /** @param {Event} evt */
  async function guarda(evt) {
   evt.preventDefault()
   try {
    if (uuid !== null) {
     const nombre =
      txtNombre.value.trim()
     const modelo = {
      uuid,
      nombre,
      eliminado: 0
     }
     await pasatiempoModifica(
      modelo)
     location.href = "index.html"
    }
   } catch (e) {
    muestraError(e);
   }
  }

  function elimina() {
   try {
    if (uuid !== null && confirm(
     txtConfirmaEliminacion())) {
     pasatiempoElimina(uuid)
     location.href = "index.html"
    }
   } catch (e) {
    muestraError(e);
   }
  }
 </script>
</body>

</html>
