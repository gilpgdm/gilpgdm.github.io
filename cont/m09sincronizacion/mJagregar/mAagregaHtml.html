 <script type="module">
  import {
   escuchaEventosAgrega
  } from "./lib/js/escuchaEventosAgrega.js"
  import {
   muestraError
  } from "./lib/js/muestraError.js"
  import {
   saltaAIndex
  } from "./lib/js/saltaAIndex.js"

  import "./js/configura.js"
  import {
   pasatiempoAgrega
  } from "./js/dao/pasatiempoAgrega.js"

  preparaVista()

  function preparaVista() {
   escuchaEventosAgrega(forma,
    agrega)
  }

  /** @param {Event} evt */
  async function agrega(evt) {
   try {
    evt.preventDefault()
    const nombre =
     forma.nombre.value.trim()
    const modelo = {
     nombre
    }
    await pasatiempoAgrega(modelo)
    saltaAIndex()
   } catch (e) {
    muestraError(e)
   }
  }
 </script>
