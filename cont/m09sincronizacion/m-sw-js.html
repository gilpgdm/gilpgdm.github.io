73// Verifica si el código corre dentro de un service worker.
74if (self instanceof ServiceWorkerGlobalScope) {
75 // Evento al empezar a instalar el servide worker,
76 self.addEventListener("install",
77 (/** @type {ExtendableEvent} */ evt) => {
78 console.log("El service worker se está instalando.")
79 evt.waitUntil(llenaElCache())
80 })
81
82 // Evento al solicitar información a la red.
83 self.addEventListener("fetch", (/** @type {FetchEvent} */ evt) => {
84 if (evt.request.method === "GET") {
85 evt.respondWith(buscaLaRespuestaEnElCache(evt))
86 }
87 })
88
89 // Evento cuando el service worker se vuelve activo.
90 self.addEventListener("activate",
91 () => console.log("El service worker está activo."))
92}

94async function llenaElCache() {
95 console.log("Intentando cargar caché:", CACHE)
96 // Borra todos los cachés.
97 const keys = await caches.keys()
98 for (const key of keys) {
99 await caches.delete(key)
100 }
101 // Abre el caché de este service worker.
102 const cache = await caches.open(CACHE)
103 // Carga el listado de ARCHIVOS.
104 await cache.addAll(ARCHIVOS)
105 console.log("Cache cargado:", CACHE)
106 console.log("Versión:", VERSION)
107}

109/** @param {FetchEvent} evt */
110async function buscaLaRespuestaEnElCache(evt) {
111 // Abre el caché.
112 const cache = await caches.open(CACHE)
113 const request = evt.request
114 /* Busca la respuesta a la solicitud en el contenido del caché, sin
115 * tomar en cuenta la parte después del símbolo "?" en la URL. */
116 const response = await cache.match(request, { ignoreSearch: true })
117 if (response === undefined) {
118 /* Si no la encuentra, empieza a descargar de la red y devuelve
119 * la promesa. */
120 return fetch(request)
121 } else {
122 // Si la encuentra, devuelve la respuesta encontrada en el caché.
123 return response
124 }
