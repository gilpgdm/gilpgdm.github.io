<?php
require_once __DIR__ .
 "/../modelo/Dispositivo.php";
 require_once __DIR__ .
 "/AccesoBd.php";

function dispositivoModifica(
 Dispositivo $dispositivo
) {
 $dispositivo->validaCambio();
 $con = AccesoBd::getCon();
 $stmt = $con->prepare(
  "UPDATE DISPOSITIVO
   SET DIS_VALOR = :valor
   WHERE DIS_ID = :id"
 );
 $stmt->execute([
  ":id" => $dispositivo->id,
  ":valor" => $dispositivo->valor
 ]);
}
