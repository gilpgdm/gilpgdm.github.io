<?php

namespace srv\dao;

require_once "srv/dao/bdCrea.php";
require_once
 "srv/dao/dispositivoBusca.php";
require_once
 "srv/dao/dispositivoAgrega.php";

use \PDO;
use srv\modelo\Dispositivo;

class AccesoBd
{

 private static ?PDO $con = null;

 static function getCon(): PDO
 {
  if (self::$con === null) {
   self::$con = self::conecta();
   self::prepara(self::$con);
  }
  return self::$con;
 }

 private static
 function conecta(): PDO
 {
  return new PDO(
   // cadena de conexión
   "sqlite:iotwss.db",
   // usuario
   null,
   // contraseña
   null,
   [PDO::ATTR_ERRMODE =>
   PDO::ERRMODE_EXCEPTION]
  );
 }
 private static
 function prepara(PDO $con)
 {
  bdCrea($con);
  $dispositivo =
   dispositivoBusca("1");
  if (!$dispositivo) {
   $dispositivo =
    new Dispositivo();
   $dispositivo->id = "1";
   $dispositivo->valor = "0";
   dispositivoAgrega($dispositivo);
  }
 }
}
