<?php
require __DIR__ .
 '/vendor/autoload.php';
require_once "./Eventos.php";
require_once
 "./dao/dispositivoBusca.php";

use Ratchet\Server\IoServer;
use Ratchet\Http\HttpServer;
use Ratchet\WebSocket\WsServer;

try {
 $dispositivo =
  dispositivoBusca("1");
 $valor = $dispositivo
  ? $dispositivo->valor : "0";
 $server = IoServer::factory(
  new HttpServer(
   new WsServer(
    new Eventos($valor)
   )
  ),
  80
 );

 $server->run();
} catch (\Throwable $e) {
 echo $e->getMessage();
}
