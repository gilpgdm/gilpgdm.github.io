<?php
function bdCrea(\PDO $con)
{
 $con->exec(
  'CREATE TABLE
   IF NOT EXISTS DISPOSITIVO (
    DIS_ID TEXT PRIMARY KEY,
    DIS_VALOR TEXT NOT NULL
   )'
 );
 $con->exec(
  'CREATE TABLE
   IF NOT EXISTS HISTORICO (
    HIS_ID INTEGER PRIMARY KEY,
    DIS_ID TEXT NOT NULL,
    HIS_VALOR TEXT NOT NULL,
    HIS_TIMESTAMP TEXT NOT NULL,
    FOREIGN KEY (DIS_ID)
     REFERENCES DISPOSITIVO(DIS_ID)
   )'
 );
}
