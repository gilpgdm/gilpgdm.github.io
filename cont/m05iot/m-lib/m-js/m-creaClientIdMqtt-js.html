/**
 * Añade caracteres al azar a una raíz, para obtener un clientId único por cada
 * instancia que se conecte al servidor de mqtt.
 * @param {string} raiz
 */
export function creaClientIdMqtt(raiz) {
 const chars = "ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789"
 for (var i = 0; i < 15; i++) {
  raiz += chars.charAt(Math.floor(Math.random() * chars.length))
 }
 return raiz
}

// Permite que los eventos de html usen la función.
window["creaClientIdMqtt"] = creaClientIdMqtt
