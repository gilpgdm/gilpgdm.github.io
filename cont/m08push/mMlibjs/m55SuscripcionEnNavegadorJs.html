import {
 getRegistroDeServiceWorker
} from "./getRegistroDeServiceWorker.js"

export class
 SuscripcionEnNavegador {

 static async getSuscripcion() {
  const registro = await
   getRegistroDeServiceWorker()
  return registro.pushManager.
   getSubscription()
 }

 /**
  * @param {Uint8Array
  *         } applicationServerKey
  */
 static async suscribe(
  applicationServerKey) {
  const registro = await
   getRegistroDeServiceWorker()
  return registro.pushManager.
   subscribe({
    userVisibleOnly: true,
    applicationServerKey
   })
 }

 static async cancela() {
  const suscripcion = await
   SuscripcionEnNavegador
    .getSuscripcion()
  const resultado =
   suscripcion === null
    ? false
    : await
     suscripcion.unsubscribe()
  return resultado === true ?
   suscripcion
   : null
 }
}
