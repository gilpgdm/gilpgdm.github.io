import {
 activaNotificaciones
} from "./activaNotificaciones.js"
import {
 registraServiceWorker
} from "./registraServiceWorker.js"
import {
 txtNoSoportaNotificaciones
} from "./txt/txtNoSoportaNotificaciones.js"
import {
 txtNoSoportaNotificacionesPush
} from "./txt/txtNoSoportaNotificacionesPush.js"
import {
 txtRegistrado
} from "./txt/txtRegistrado.js"

/**
 * @param {string | URL} sw URL del
 *  service worker que recibe la
 *  notificaciones push.
 */
export async function
 activaNotificacionesPush(sw) {
 validaQueUsaNotificacionesPush()
 activaNotificaciones()
 const registro = await
  registraServiceWorker(sw)
 console.log(sw, txtRegistrado())
 console.log(registro)
 if (!("showNotification"
  in registro))
  throw new Error(
   txtNoSoportaNotificaciones())
}

function
 validaQueUsaNotificacionesPush() {
 if (!('PushManager' in window)) {
  throw new Error(
   txtNoSoportaNotificacionesPush()
  )
 }
}
