import {
 ejecuta
} from "./ejecuta.js"
import {
 MetodoHttp
} from "./MetodoHttp.js"

/**
* @param {Object} datos
* @param {MetodoHttp} metodoHttp
* @param {string} url
*/
export async function enviaJson
 (datos, metodoHttp, url) {
 return await ejecuta(fetch(url,
  {
   method: metodoHttp,
   headers: {
    "Content-Type":
     "application/json"
   },
   body: JSON.stringify(datos)
  }))
}
