import {
 validaQueUsaServiceWorkers
} from "./validaQueUsaServiceWorkers.js"

/**
 * @param {string | URL} sw URL del
 *  service worker que recibe la
 *  notificaciones push.
 */
export function
 registraServiceWorker(sw) {
 validaQueUsaServiceWorkers()
 return navigator.
  serviceWorker.
  register(sw)
}
