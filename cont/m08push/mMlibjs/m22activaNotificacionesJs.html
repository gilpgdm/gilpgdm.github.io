import {
 getPermisoParaUsarNotificaciones
} from "./getPermisoParaUsarNotificaciones.js"
import {
 pideAutorizarNotificaciones
} from "./pideAutorizarNotificaciones.js"
import {
 txtNotificacionesBloqueadas
} from "./txt/txtNotificacionesBloqueadas.js"
import {
 validaQueUsaNotificaciones
} from "./validaQueUsaNotificaciones.js"

export async function
 activaNotificaciones() {
 validaQueUsaNotificaciones()
 let permiso =
  getPermisoParaUsarNotificaciones()
 if (permiso === "default") {
  // Permiso no asignado.
  permiso = await
   pideAutorizarNotificaciones()
 }
 if (permiso === "denied") {
  // Permiso rechazado.
  throw new Error(
   txtNotificacionesBloqueadas())
 }
}
