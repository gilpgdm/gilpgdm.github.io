96
97 constructor() {
98 super()
99 if (ES_APPLE) {
100 document.body.classList.add("apple")
101 document.body.classList.remove("material")
102 } else {
103 document.body.classList.add("material")
104 document.body.classList.remove("apple")
105 }
106
107 /**
108 * @private
109 * @readonly
110 */
111 const shadow = this.attachShadow({ mode: "open" })
112 shadow.innerHTML = this.getContent()
113 /**
114 * @private
115 * @type {number}
116 */
117 this._posY = 0
118 /**
119 * @private
120 * @type {boolean}
121 */
122 this._scrolling = false
123 /**
124 * @private
125 * @type { HTMLSlotElement }
126 */
127 this._navigation = querySelector(shadow, '[name="navigation"]')
128 /**
129 * @private
130 * @type { HTMLSlotElement }
131 */
132 this._action = querySelector(shadow, '[name="action"]')
133 /**
134 * @private
135 * @type { HTMLHeadingElement | null }
136 */
137 this._headline = null
138 /**
139 * @private
140 * @type { HTMLElement | null }
141 */
142 this._adicional = null
143 addEventListener("scroll", () => this._onScroll())
144 addEventListener("load", () => this.configurOtros())
145 }
146
147 connectedCallback() {
148 this.role = "toolbar"
149 this.configuraAction()
150 }
151
152 configurOtros() {
153 const idHeadline = getAttribute(this, "headline")
154 if (idHeadline !== ) {
155 const headline = document.getElementById(idHeadline)
156 if (headline instanceof HTMLHeadingElement) {
157 this._headline = headline
158 this.classList.add("headline")
159 if (this.classList.contains("apple") || this.classList.contains("medium")) {
160 headline.classList.add("md-headline", "headline-small")
161 } else {
162 headline.classList.add("md-headline", "headline-medium")
163 }
164 }
165 }
166 const idAdicional = getAttribute(this, "adicional")
167 if (idAdicional !== ) {
168 this._adicional = document.getElementById(idAdicional)
169 if (this._adicional !== null) {
170 if (this.classList.contains("apple")) {
171 this._adicional.style.top = "env(titlebar-area-height, 48px)"
172 } else {
173 this._adicional.style.top = "env(titlebar-area-height, 64px)"
174 }
175 }
176 }
177 }
178
179 configuraAction() {
180 const assignedElements = this._action.assignedElements()
181 if (this.isConnected) {
182 if (ES_APPLE) {
183 this.classList.add("apple")
184 this.classList.remove("material")
185 } else {
186 this.classList.add("material")
187 this.classList.remove("apple")
188 }
189 if (this.classList.contains("center-aligned")) {
190 this.classList.remove("centrado")
191 this.classList.remove("justificado")
192 } else {
193 if (ES_APPLE && assignedElements.length <= 1) {
194 this.classList.add("centrado")
195 this.classList.remove("justificado")
196 } else {
197 this.classList.add("justificado")
198 this.classList.remove("centrado")
199 }
200 }
201 }
202 }
203
204 /** @private */
205 _onScroll() {
206 this._posY = scrollY
207 if (!this._scrolling) {
208 requestAnimationFrame(() => this._avanza())
209 }
210 this._scrolling = true
211 }
212
213 /** @private */
214 _avanza() {
215 if (this._posY === 0) {
216 this.classList.remove("scroll")
217 if (this._headline !== null) {
218 if (this._adicional === null) {
219 this._headline.classList.remove("scroll")
220 } else {
221 this._headline.classList.remove("scroll-adicional")
222 }
223 }
224 if (this._adicional !== null) {
225 this._adicional.classList.remove("scroll")
226 }
227 } else {
228 this.classList.add("scroll")
229 if (this._headline !== null) {
230 if (this._adicional === null) {
231 this._headline.classList.add("scroll")
232 } else {
233 this._headline.classList.add("scroll-adicional")
234 }
235 }
236 if (this._adicional !== null) {
237 this._adicional.classList.add("scroll")
238 }
239 }
240 this._scrolling = false
241 }
242
243}
244
245customElements.define("md-top-app-bar", MdTopAppBar)
