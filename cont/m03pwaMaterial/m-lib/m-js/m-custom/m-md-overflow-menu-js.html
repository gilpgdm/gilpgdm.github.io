1import { abreElementoHtml } from "../abreElementoHtml.js"
2import { cierraElementoHtmo } from "../cierraElementoHtmo.js"
3import { html } from "../html.js"
4
5export class MdOverflowMenu extends HTMLElement {
6
7 getContent() {
8 return html`
9
10 <style>
11
12 :host {
13 position: fixed;
14 min-width: 112px;
15 max-width: 280px;
16 }
17
18 ::slotted(*) {
19 text-align: start;
20 width: 100%;

22 background-color: transparent;
23 }
24
25 </style>
26
27 <slot></slot>`.htmlTxt
28 }
29
30 constructor() {
31 super()
32 const shadow = this.attachShadow({ mode: "open" })
33 shadow.innerHTML = this.getContent()
34 this.clicCierra = this.clicCierra.bind(this)
35 /**
36 * @private
37 * @type {HTMLButtonElement| null}

39 this._toggleButton = null
40 }

42 connectedCallback() {

46 }
47
48 /**
49 * @param {HTMLButtonElement} toggleButton
50 */
51 alterna(toggleButton) {
52 this._toggleButton = toggleButton
53 const top = toggleButton.offsetTop + toggleButton.offsetHeight - 4
54 const right =
55 innerWidth - (toggleButton.offsetLeft + toggleButton.offsetWidth) - 3
56 this.style.top = `${top}px`
57 this.style.right = `${right}px`
58 const list = this.classList
59 if (list.contains("open")) {
60 this.cierra()
61 } else {
62 this.abre()
63 }
64 }
65
66 abre() {
67 document.addEventListener("click", this.clicCierra)
68 abreElementoHtml(this)
69 }
70
71 cierra() {
72 document.removeEventListener("click", this.clicCierra)
73 cierraElementoHtmo(this)
74 }
75
76 /**
77 * @param {Event} evt
78 */
79 clicCierra(evt) {
80 const target = evt.target
81 if (this.classList.contains("open")
82 && this._toggleButton !== null
83 && target instanceof HTMLElement
84 && !this._toggleButton.contains(target)) {
85 this.cierra()
86 }
87 }
88}
89
90customElements.define("md-overflow-menu", MdOverflowMenu)
