3/**
4 * @param { HTMLElement } elementoHtmlQueInvoca
5 * @param { string } nombreDelMetodo
6 * @param { Event } event
7 */
8export function ejecutaMetodo(elementoHtmlQueInvoca, nombreDelMetodo, event) {
9 try {
10 let nodoPadre = getNodoPadre(elementoHtmlQueInvoca)
11 while (nodoPadre !== null) {
12 const metodo = nodoPadre[nombreDelMetodo]
13 if (typeof metodo === "function") {
14 metodo.call(nodoPadre, event)
15 break
16 }
17 nodoPadre = getNodoPadre(nodoPadre)
18 }
19 if (nodoPadre === null) {
20 const metodo = window[nombreDelMetodo]
21 if (typeof metodo === "function") {
22 metodo.call(window, event)
23 }
24 }
25 } catch (error) {
26 muestraError(error)
27 }
28}
29
30/**
31 * @param { Node } nodo
32 */
33function getNodoPadre(nodo) {
34 if (nodo instanceof ShadowRoot) {
35 return nodo.host
36 } else {
37 return nodo.parentNode
38 }
39}
