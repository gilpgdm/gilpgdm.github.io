130// Verifica si el código corre dentro de un service worker.
131if (self instanceof ServiceWorkerGlobalScope) {
132 // Evento al empezar a instalar el servide worker,
133 self.addEventListener("install",
134 (/** @type {ExtendableEvent} */ evt) => {
135 console.log("El service worker se está instalando.")
136 evt.waitUntil(llenaElCache())
137 })
138
139 // Evento al solicitar información a la red.
140 self.addEventListener("fetch", (/** @type {FetchEvent} */ evt) => {
141 if (evt.request.method === "GET") {
142 evt.respondWith(buscaLaRespuestaEnElCache(evt))
143 }
144 })
145
146 // Evento cuando el service worker se vuelve activo.
147 self.addEventListener("activate",
148 () => console.log("El service worker está activo."))
149}

151async function llenaElCache() {
152 console.log("Intentando cargar caché:", CACHE, ".")
153 // Borra todos los cachés.
154 const keys = await caches.keys()
155 for (const key of keys) {
156 await caches.delete(key)
157 }
158 // Abre el caché de este service worker.
159 const cache = await caches.open(CACHE)
160 // Carga el listado de ARCHIVOS.
161 await cache.addAll(ARCHIVOS)
162 console.log("Cache cargado:", CACHE, ".")
163 console.log("Versión:", VERSION, ".")
164}
165
166/** @param {FetchEvent} evt */
167async function buscaLaRespuestaEnElCache(evt) {
168 // Abre el caché.
169 const cache = await caches.open(CACHE)
170 const request = evt.request
171 /* Busca la respuesta a la solicitud en el contenido del caché, sin
172 * tomar en cuenta la parte después del símbolo "?" en la URL. */
173 const response = await cache.match(request, { ignoreSearch: true })
174 if (response === undefined) {
175 /* Si no la encuentra, empieza a descargar de la red y devuelve
176 * la promesa. */
177 return fetch(request)
178 } else {
179 // Si la encuentra, devuelve la respuesta encontrada en el caché.
180 return response
181 }
