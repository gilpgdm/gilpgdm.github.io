import {
 LABEL_NAME, SUPPORTING_NAME,
 TARGET_NAME
} from "../constantes.js"

export class SliderField
 extends HTMLElement {

 constructor() {
  super()
  const shadow =
   this.attachShadow({
    mode: "open",
    delegatesFocus: true
   })

  shadow.innerHTML = /* html */
   `<style>
     :host {
      display: block;
      margin: 1rem;
     }

     label {
      display: block;
     }

     ::slotted(input) {
      vertical-align: middle;
      -webkit-appearance: none;
      background-image:
       linear-gradient(to right, var(--Primary) 0%, var(--Primary) 50%, var(--SurfaceVariant) 50%, var(--SurfaceVariant) 100%);
      border-radius: 2px;
      height: 4px;
     }

     [${LABEL_NAME}]::slotted(*) {
      display: block;
      color:
       var(--OnSurfaceVariant);
      font-family:
       var(--BodyLargeFont);
      font-size:
       var(--BodySmallSize);
      font-weight:
       var(--BodyLargeWeight);
      margin-bottom: 8px;
     }

     [${TARGET_NAME}]::slotted(input:focus) {
      outline: none;
     }

     [${SUPPORTING_NAME}]::slotted(*) {
      color:
       var(--OnSurfaceVariant);
      font-family:
       var(--BodySmallFont);
      font-size:
       var(--BodySmallSize);
      font-weight:
       var(--BodySmallWeight);
      padding-top: 4px;
      display: block;
      transition: 0.3s ease-in;
     }
    </style>
    <label>
     <slot ${LABEL_NAME}></slot>
     <slot ${TARGET_NAME}></slot>
    <slot
      ${SUPPORTING_NAME}></slot>
  </label>`

  this.analiza =
   this.analiza.bind(this)
  this._configSlot =
   this._configSlot.bind(this)

  /**
  * @private
  * @type {HTMLSlotElement|null}
  */
  this._target = shadow.
   querySelector(
    `[${TARGET_NAME}]`)
  /**
   * @private
   * @type {HTMLInputElement|null}
   */
  this._input = null
  if (this._target !== null) {
   this._target.addEventListener(
    "slotchange",
    this._configSlot)
  }
 }

 /** @private */
 _configSlot() {
  if (this._input !== null) {
   this._input.removeEventListener(
    "input", this.analiza)
   this._input = null
  }
  if (this._target !== null) {
   for (const t of this._target
    .assignedElements()) {
    if (t
     instanceof HTMLInputElement) {
     this._input = t
     t.addEventListener("input",
      this.analiza)
     this.analiza()
    }
   }
  }
 }

 disconnectedCallback() {
  if (this._input !== null) {
   this._input.removeEventListener(
    "input", this.analiza)
  }
 }

 analiza() {
  const i = this._input
  if (i) {
   const v = i.valueAsNumber
   const min = parseFloat(i.min)
   const max = parseFloat(i.max)
   const value =
    (v - min) / (max - min) * 100
   i.style.background =
    `linear-gradient(to right, var(--Primary) 0%, var(--Primary) ${value
    }%, var(--SurfaceVariant) ${value
    }%, var(--SurfaceVariant) 100%)`
  }
 }

}

customElements.define(
 "slider-field", SliderField)
