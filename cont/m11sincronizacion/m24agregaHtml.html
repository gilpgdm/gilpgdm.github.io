<!DOCTYPE html>
<html lang="es">

<head>
 <meta charset="UTF-8">
 <meta name="viewport"
   content="width=device-width">
 <title>Agregar</title>
</head>

<body>
 <h1>Agregar</h1>
 <form id="form">
  <p>
   <label>
    Nombre
    <input id="txtNombre" required>
   </label>
  </p>
  <p>
   <a
     href="index.html">Cancelar</a>
   <button>Agregar</button>
  </p>
 </form>
 <script type="module">
  import {
   muestraError
  } from "./lib/js/muestraError.js"
  import "./js/config.js"
  import {
   pasatiempoAgrega
  } from "./js/dao/pasatiempoAgrega.js"

  let secuencia = 0

  form.addEventListener("submit",
   guarda)

  /** @param {Event} evt */
  async function guarda(evt) {
   try {
    evt.preventDefault()
    const nombre =
     txtNombre.value.trim()
    const uuid = Date.now().
     toString() + Math.random() +
     secuencia
    secuencia++
    const modelo = {
     uuid,
     nombre
    };
    await pasatiempoAgrega(modelo)
    location.href = "index.html"
   } catch (e) {
    muestraError(e)
   }
  }
 </script>
</body>

</html>
