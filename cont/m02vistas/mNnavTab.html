    super()
    /** @type {number} */
    this.posY = 0
    /** @type {boolean} */
    this.procesando = false
   }
   connectedCallback() {
    window.addEventListener(
     "scroll",
     () => this.onScroll())
   }
   onScroll() {
    this.posY = window.scrollY
    if (!this.procesando) {

      () => this.avanza())
    }
    this.procesando = true
   }
   avanza() {
    if (this.posY === 0) {
     this.classList.
      remove(CLASE_ELEVACION2)
    } else {
     this.classList.
      add(CLASE_ELEVACION2)
    }
    this.procesando = false
   }
