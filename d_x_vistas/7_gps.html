    class CampoDinámico
      extends HTMLElement {
      connectedCallback() {
        setTimeout(() => {
          this.prepara();
        }, 100);
      }
      prepara() {
        /**
         * @type {
            HTMLInputElement|
            HTMLTextAreaElement|
            HTMLSelectElement|
            HTMLOutputElement|
            null} */
        this._control =
          this.querySelector(
            `input,
            textarea,
            select,
            output`);
        this.addEventListener(
          "focus",
          () => this.analiza(),
          true);
        this.addEventListener(
          "focusout",
          () => this.analiza(),
          true);
        this.analiza();
      }
      analiza() {
        if (this._control) {
          if (!this._control.
            validity.badInput &&
            this._control.value.
              length === 0) {
            this._control.
              classList.
              remove("valor");
          } else {
            this._control.
              classList.
              add("valor");
          }
        }
      }
    }
