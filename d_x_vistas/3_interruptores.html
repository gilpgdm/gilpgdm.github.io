    class CampoInterruptor
      extends HTMLElement {
      connectedCallback() {
        setTimeout(
          () => this.prepara(),
          100);
      }
      prepara() {
        /** @private */
        this._input =
          this.querySelector(
            "input");
        if (this._input) {
          this._input.type =
            "range";
          this._input.min = "0";
          this._input.max = "1";
          this._input.
            addEventListener(
              "change", () =>
              this.
                analiza());
          this.analiza();
        }
      }
      analiza() {
        if (this._input) {
          if (this._input.
            valueAsNumber === 0) {
            this._input.classList.
              add("off");
          } else {
            this._input.classList.
              remove("off");
          }
        }
      }
    }
