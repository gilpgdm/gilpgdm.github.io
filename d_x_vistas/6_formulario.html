  <script>
    const USER_AGENT =
      navigator.userAgent ||
      navigator.vendor ||
      window["opera"];
    if (/android/i.
      test(USER_AGENT)) {
      document.body.classList
        .add("android");
    }
    if (/iPad|iPhone|iPod|Mac/.

    const REQUERIDO = "*Requerido";
    const REQUERIDO_NO_0 =
      "*Requerido, no 0";
    const NO_0 = "No puede ser 0";
    function validaA() {
      msg(inA, msgA, REQUERIDO);
    }
    function validaB() {
      if (inB) {
        const b =
          inB.valueAsNumber;
        if (isNaN(b) || b !== 0) {
          inB.setCustomValidity(
            "");
        } else {
          inB.setCustomValidity(
            NO_0);
        }
        msg(inB, msgB,
          REQUERIDO_NO_0);
      }
    }
    function divide() {
      try {
        if (inA.validity.valid &&
          inB.validity.valid) {
          const a =
            inA.valueAsNumber;
          const b =
            inB.valueAsNumber;
          outSalida.value = a / b;
        } else {
          outSalida.value = "";
        }
      } catch (e) {
        console.error(e);
        outSalida.value =
          e.message;
      }
    }
    /** Sin un elemento tiene un
     * mensaje de validación, lo
     * muestra en su elemento de
     * ayuda y si no, muestra un
     * texto de ayuda. 
     * @param {HTMLInputElement|
     *  HTMLTextAreaElement|
     *  HTMLSelectElement|
     *  HTMLOutputElement|null} e
     * referencia a un elemento
     * que contiene datos.
     * @param {HTMLOutputElement|
     *  null} out
     * elemento de ayuda para e.
     * @param {string} ok mensaje
     * de ayuda cuando el estado
     * de e es válido. */
    function
      msg(e, out, ok) {
      if (e && out) {
        if (e.validity.valid) {
          out.value = ok;
        } else {
          out.value =
            e.validationMessage;
        }
      }
    }
    class CampoDinámico
      extends HTMLElement {
      connectedCallback() {
        setTimeout(() => {
          this.prepara();

      }
      prepara() {
        /**
         * @type {
            HTMLInputElement|
            HTMLTextAreaElement|
            HTMLSelectElement|
            HTMLOutputElement|
            null} */
        this._control =
          this.querySelector(
            `input,
            textarea,
            select,
            output`);
        this.addEventListener(
          "focus",
          () => this.analiza(),
          true);
        this.addEventListener(
          "focusout",
          () => this.analiza(),
          true);
        this.analiza();
      }
      analiza() {
        if (this._control) {
          if (!this._control.
            validity.badInput &&
            this._control.value.
              length === 0) {
            this._control.
              classList.
              remove("valor");
          } else {
            this._control.
              classList.
              add("valor");
          }
        }
      }
    }
