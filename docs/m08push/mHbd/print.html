{

    SUS_ENDPOINT TEXT NOT NULL,
    SUS_PUB_KEY TEXT NOT NULL,
    SUS_AUT_TOK TEXT NOT NULL,
    SUS_CONT_ENCOD TEXT NOT NULL,

   )'
 );

3. srv / dao / accesobd.php

<?php

namespace srv\dao;

require_once "srv/dao/bdCrea.php";

use \PDO;

class AccesoBd
{

 private static ?PDO $con = null;

 static function getCon(): PDO
 {
  if (self::$con === null) {
   self::$con = self::conecta();
   self::prepara(self::$con);
  }
  return self::$con;
 }

 private static
 function conecta(): PDO
 {
  return new PDO(
   // cadena de conexión
   "sqlite:notificacionespush.db",
   // usuario
   null,
   // contraseña
   null,
   [PDO::ATTR_ERRMODE =>
   PDO::ERRMODE_EXCEPTION]
  );
 }

 private static
 function prepara(PDO $con)
 {
  bdCrea($con);
 }
}

4. srv / dao / suscripcionAgrega.php

<?php

use srv\dao\AccesoBd;
use srv\modelo\Suscripcion;

function suscripcionAgrega(
 Suscripcion $modelo
) {
 $con = AccesoBd::getCon();
 $stmt = $con->prepare(
  "INSERT INTO SUSCRIPCION
    (SUS_ENDPOINT, SUS_PUB_KEY,
    SUS_AUT_TOK, SUS_CONT_ENCOD)
   VALUES
    (:endpoint, :publicKey,
     :authToken, :contentEncoding)"
 );
 $stmt->execute([
  ":endpoint" => $modelo->endpoint,
  ":publicKey"
 => $modelo->publicKey,
  ":authToken"
 => $modelo->authToken,
  ":contentEncoding"
  => $modelo->contentEncoding
 ]);
}

5. srv / dao / suscripcionBusca.php

<?php

use srv\dao\AccesoBd;
use srv\modelo\Suscripcion;

function suscripcionBusca(
 string $endpoint
) {
 $con = AccesoBd::getCon();
 $stmt = $con->prepare(
  "SELECT
    SUS_ENDPOINT as endpoint,
    SUS_PUB_KEY as publicKey,
    SUS_AUT_TOK as authToken,
    SUS_CONT_ENCOD as
      contentEncoding
   FROM SUSCRIPCION
   WHERE SUS_ENDPOINT = :endpoint"
 );
 $stmt->execute(
  [":endpoint" => $endpoint]
 );
 $stmt->setFetchMode(
  PDO::FETCH_CLASS,
  Suscripcion::class
 );
 return $stmt->fetch();
}
