C. Instrucciones

  1. Este ejemplo requiere 2 proyectos

  2. Revisa el proyecto en phpws en Replit con la URl https://replit.com/@GilbertoPacheco/phpws?v=1

  3. Revisa el proyecto iot2 en Replit con la URl https://replit.com/@GilbertoPacheco/iot2?v=1

  4. iot2 puede correr simultáneamente en varios navegadores y computadoras, todos interactuando con la misma instancia de phpws.

  5. Usa o crea una cuenta de Google.

  6. Crea una cuenta de Replit usando la cuenta de Google.

  7. Crea el proyecto phpws con la categoría PHP CLI en Replit y edita o sube los archivos de este proyecto.

  8. Para el contenido de la carpeta vendor y los archivos composer tienes 2 opciones:

    • Descargar el siguiente archivo zip: composer.zip.

    • Crear una carpeta, usar un shell como CMD, sh o PowerShell en esa carpeta, tener PHP y composer para PHP instalados en tu computadora y usar la orden:

      composer require cboden/ratchet

  9. Depura el proyecto.

  10. Crea la página de spotlight del proyecto.

  11. Crea el proyecto iot2 con la categoría HTML, CSS, JS en Replit y edita o sube los archivos de este proyecto.

  12. Depura el proyecto.

  13. Crea la página de spotlight del proyecto.

K. dao/bdCrea.php

<?php
function bdCrea(\PDO $con)
{
 $con->exec(
  'CREATE TABLE
   IF NOT EXISTS DISPOSITIVO (
    DIS_ID TEXT PRIMARY KEY,
    DIS_VALOR TEXT NOT NULL
   )'
 );
 $con->exec(
  'CREATE TABLE
   IF NOT EXISTS HISTORICO (
    HIS_ID INTEGER PRIMARY KEY,
    DIS_ID TEXT NOT NULL,
    HIS_VALOR TEXT NOT NULL,
    HIS_TIMESTAMP TEXT NOT NULL,
    FOREIGN KEY (DIS_ID)
     REFERENCES DISPOSITIVO(DIS_ID)
   )'
 );
}

L. dao/AccesoBd.php

<?php
require_once __DIR__ .
 "/bdCrea.php";
require_once __DIR__ .
 "/dispositivoBusca.php";
require_once __DIR__ .
 "/dispositivoAgrega.php";

class AccesoBd
{
 private static $con;

 public static
 function getCon(): \PDO
 {
  if (!self::$con) {
   self::$con = self::conecta();
   self::prepara();
  }
  return self::$con;
 }
 public static
 function conecta(): \PDO
 {
  return new PDO(
   "sqlite:phpws.db",
   null,
   null,
   [PDO::ATTR_ERRMODE =>
   PDO::ERRMODE_EXCEPTION]
  );
 }
 public static function prepara()
 {
  bdCrea(self::getCon());
  $dispositivo =
   dispositivoBusca("1");
  if (!$dispositivo) {
   $dispositivo =
    new Dispositivo();
   $dispositivo->id = "1";
   $dispositivo->valor = "0";
   dispositivoAgrega($dispositivo);
  }
 }
}

  "SELECT
     DIS_ID as id,
     DIS_VALOR asvalor
   FROM DISPOSITIVO
   WHERE DIS_ID = :id"
 );

  ":id" => $id
 ]);
 $stmt->setFetchMode(
  PDO::FETCH_CLASS,
  "Dispositivo"
 );

}

S. Eventos.php

<?php
require_once "./guarda.php";

use Ratchet\MessageComponentInterface;
use Ratchet\ConnectionInterface;

class Eventos implements
 MessageComponentInterface
{
 private $valor;
 protected $clientes;

 public function __construct(
  $valor
 ) {
  $this->valor = $valor;
  $this->clientes =
   new \SplObjectStorage;
 }

 public function onOpen(
  ConnectionInterface $con
 ) {
  $this->clientes->attach($con);
  $con->send($this->valor);
  echo "Conectado: " .
   $con->resourceId . "\n";
 }

 public function onMessage(
  ConnectionInterface $from,
  $msg
 ) {
  echo "Recibido: $msg\n";
  $this->valor =
   $this->valor === "0"
   ? "1" : "0";
  guarda($this->valor);
  foreach ($this->clientes
   as $client) {
   $client->send($this->valor);
  }
 }

 public function onClose(
  ConnectionInterface $con
 ) {
  $this->clientes->detach($con);
  echo "Desconectado: " .
   $con->resourceId . "\n";
 }

 public function onError(
  ConnectionInterface $con,
  \Exception $e
 ) {
  echo "Error: " .
   $e->getMessage() . "\n";
  $con->close();
 }
}

T. main.php

<?php
require __DIR__ .
 '/vendor/autoload.php';
require_once "./Eventos.php";
require_once
 "./dao/dispositivoBusca.php";

use Ratchet\Server\IoServer;
use Ratchet\Http\HttpServer;
use Ratchet\WebSocket\WsServer;

try {
 $dispositivo =
  dispositivoBusca("1");
 $valor = $dispositivo
  ? $dispositivo->valor : "0";
 $server = IoServer::factory(
  new HttpServer(
   new WsServer(
    new Eventos($valor)
   )
  ),
  80
 );

 $server->run();
} catch (\Throwable $e) {
 echo $e->getMessage();
}
