A. App nativa

App desarrollada específicamente para una plataforma de cómputo.

Ejemplos de Plataformas

  • Web
  • Android
  • Windows
  • iOS (iPad y iPhone)
  • MacOS

B. App multiplataforma

App diseñada para ejecutarse en diferentes plataformas.

Tecnologías Multiplataforma

  • Progressive Web Application (PWA)
  • Cordova
  • Native Script
  • Ionic

C. Diferencias entre app nativa y app multiplataforma

App Nativa App Multiplataforma
1 código por plataforma. (Posiblemente en distintos lenguajes.) 1 código único para todas las plataformas.

D. Ventajas de las app nativas

  • Permiten que la experiencia de usuario se integre mejor al uso de la plataforma.

  • Mejor rendimiento.

  • Sacan mejor provecho a las características de la plataforma.

E. Desventajas de las app nativas

  • Requieren un equipo de desarrollo con más conocimientos.

  • Hay que desarrollar y dar mantenimiento a más código.

  • Más caro.

  • Más tiempo para desarrollar.

  • Forma de trabajo más complicada.

  • Software que requiere más recursos de hardware.

F. Ventajas de las app multiplataforma

  • 1 solo código.

  • Menor tiempo de desarrollo.

  • Desarrollo más barato.

G. Desventajas de las app multiplataforma

  • Menor rendimiento.

  • Se integran menos con el sistem operativo.

B. Un formulario para móvil

  • Los formularios móviles muestran los mensajes de error del componente en otro componente conocido como texto de ayuda y que en este ejemplo se muestra con el id msgGenero.

  • En el resto de la lección se muestra como dar apariencia móvil a la interfaz gráfica.

 <script>
  const REQUERIDO = "*Requerido"

  function procesa() {
   try {
    if (inGenero.validity.valid) {
     const genero = inGenero.value
     if (genero === "pop") {
      outSalida.value =
       "Escucha a Ariana Grande."
     } else if (genero === "reg") {
      outSalida.value =
       "Escucha a Bad Bunny."
     } else {
      outSalida.value =
       "De ese género no conozco."
     }
    } else {
     outSalida.value = ""
    }
   } catch (e) {
    console.error(e)
    outSalida.value = e.message
   }
  }

  function copiaMensajes() {
   msg(inGenero, msgGenero,
    REQUERIDO)
  }

  /** Si un elemento tiene un
   * mensaje de validación, lo
   * muestra en su elemento de
   * ayuda y si no, muestra un
   * texto de ayuda.
   * @param {HTMLInputElement|
   *  HTMLTextAreaElement|
   *  HTMLSelectElement|
   *  HTMLOutputElement|null} e
   *   referencia a un elemento
   *   que contiene datos.
   * @param {HTMLOutputElement|
   *  null} out  referencia a un
   *   elemento de ayuda para e.
   * @param {string} ok mensaje de
   *   ayuda cuando el estado de e
   *   es válido. */
  function msg(e, out, ok) {
   if (e && out) {
    if (e.validity.valid) {
     out.value = ok
    } else {
     out.value =
      e.validationMessage
    }
   }
  }
 </script>
</body>
</html>

D. Campos

  • Los campos de texto en Material Design están definidas en https://material.io/components/text-fields.

  • Los menús en Material Design están definidos en https://material.io/components/menus.

  • Se usa una etiqueta flotante que se muestra grande cuando el contenido de campo es una cadena vacía y se muestra pequeña cuando el elemento está en uso o cuando su contenido no está vacío.

  • Para que el contenido se muestra adecuadamente, se define el elemento personalizado cm-din y su comportamiento lo definimos con la clase CmDin localizada al final del código.

const CUE = "cue"

}

actualizaFormas()

function actualizaFormas() {
 const cue =
  sessionStorage.getItem(CUE)
 if (sesion !== null) {
  if (cue === null) {
   // Sesión terminada.
   sesion.hidden = true
   sesion.cue.value = ""
  } else {
   // Sesión iniciada.
   sesion.hidden = false
   sesion.cue.value = cue
  }
 }
 if (login !== null) {
  login.cue.value = ""
  login.match.value = ""
  login.humano.valueAsNumber = 0
  esHumano = false;
  login.hidden = cue !== null
 }
}
/** @param {Event} evt */
async function iniciaSesión(evt) {
 try {
  evt.preventDefault();
  if (login !== null
   && sesion !== null) {
   if (!esHumano) {
    throw new Error(
     "Comprueba que eres humano.")
   }
   // Toma los datos de la forma.
   const datos =
    new FormData(login)
   /* Usa el método POST para
    * enviar los datos de la forma
    * al recurso
    *  "/valida"
    * en el servidor. */
   const res = await fetch(
    "/valida",
    {
     method: "POST",
     body: datos
    })
   if (res.ok) {
    const resp = await res.text()
    if (resp === "ok") {
     sessionStorage.setItem(CUE,
      login.cue.value)
     actualizaFormas()
    } else {
     alert("Datos incorrectos.")
    }
   } else {
    throw new Error(res.statusText)
   }
  }
 } catch (e) {
  console.error(e)
  alert(e.message)
 }
}

async function terminaSesión() {
 sessionStorage.removeItem(CUE)
 actualizaFormas()
}

/**
 * @param {import("express").
 *   Request} req solicitud que
 *    recibe el servidor web.
 * @param {import("express").
 *  Response} res respuesta que
 *    devuelve el servidor web. */
async function
 validaUsuario(req, res) {
 /* Crea un cliente que se conecta
  * a la base de datos. */
 const client = new MongoClient(
  uri,
  {
   useNewUrlParser: true,
   useUnifiedTopology: true
  })
 try {
  await client.connect()
  const usuario = client.
   db("aut").
   collection("usuario")
  /* Los datos enviados por la
   * página vienen en la propiedad
   * body de req. */
  const datos = req.body
  /* Cuenta en la base de datos
   * los documentos que tengan
   *  nombre === datos.cue
   * y
   *  contrasena === datos.match
   */
  const total = await usuario.
   countDocuments({
    nombre: datos.cue,
    contrasena: datos.match
   })
  if (total > 0) {
   /* Devuelve a la página el texto
    *  "ok". */
   res.send("ok")
  } else {
   /* Devuelve a la página el texto
    *  "". */
   res.send("")
  }
 } catch (e) {
  res.send("")

A. Blink.ino

Para empezar a programar, revisa que al conectar tu dispositivo ESP8266, Arduino IDE lo reconozca y tenga habilitado el menú Herramientas > Puerto; si no es así, tienes que revisar que el driver para Windows esté bien cargado.

Revisa que el driver para Windows esté bien cargado.

Ve como capturar y ejecutar un ejemplo parecido a este.

Ve como desplegar este ejemplo en Tinkercad y compartirlo.

/* Este programa es un derivado de
   ESP8266 Blink by Simon Peter */

/** Función que se invoca una sola
    vez al inicio del programa. */
void setup() {
  /* Todas las placas compatibles
     con Arduino incluyen un led
     de prueba ligado a un pin
     cuyo número está definido por
     la constante LED_BUILTIN.
     Inicializa el pin para
     LED_BUILTIN como salida. */
  pinMode(LED_BUILTIN, OUTPUT);
}

/* Función que se invoca
   repetidamente mientras el
   programa esté activo. */
void loop() {
  /* Enciende el LED_BUILTIN (LOW
     es el nivel de voltaje). */
  digitalWrite(LED_BUILTIN, LOW);
  delay(1000); // Espera 1 segundo
  /* Apaga el LED_BUILTIN haciendo
     que el nivel de voltaje sea
     HIGH. */
  digitalWrite(LED_BUILTIN, HIGH);
  delay(1000);
}

B. Button.ino

Ve como desplegar este ejemplo en Tinkercad y compartirlo.

/* Este es un derivado de Button
   por DojoDave y Tom Igoe */

/* Pin para el botón. */
const int BOTON = 0;

int estadoDelBoton = 0;

void setup() {
  /* Inicializa el pin para
     LED_BUILTIN como salida. */
  pinMode(LED_BUILTIN, OUTPUT);
  /* Initializa el pin para
     BOTON como entrada. */
  pinMode(BOTON, INPUT);
}

void loop() {
  estadoDelBoton =
    digitalRead(BOTON);
  /* Checa si el botón está
     presionado. Esto es, si
     estadoDelBotón es LOW: */
  if (estadoDelBoton == LOW) {
    // Enciende el LED_BUILTIN.
    digitalWrite(LED_BUILTIN,
      LOW);
  } else {
    // Apaga el LED_BUILTIN.
    digitalWrite(LED_BUILTIN,
      HIGH);
  }
}

C. Dispositivo.ino

  • En el siguiente código, sustituye la configuración por los datos de tu red WiFi. La URL y la huella digital deben correspponder a tu proyecto de IoT.

  • Las siguientes instrucciones son para obtener la huella digital:

    1. Usa Chrome para abrir la URL del código de la lección IoT. En este caso es https://iot.gilbertopachec3.repl.co/.

    2. Haz clic en el candadito que está junto a la URL. Se abre un menú contextual.

    3. Haz clic en donde dice Certificado (válido). Aparece un cuadro de diálogo. Observa hasta cuando puedes usar esta información. Despues de la fecha indicada, debes actualizar tu código.

    4. Selecciona la pestaña Detalles. Haz scroll hasta que aparezca la Huella digital. Haz clic en esta propiedad y abajo se muestra su valor. Copia y pega este valor a tu código.

  • Para monitorear su funcionamiento, debes hacer clic en el botón de arriba a la derecha en la ventana de Arduino, que tiene una lupa y tiene el mensaje flotante que dice Monitor Serie. Debes ajustar los baudios a la misma velocidad que en el código, donde dice Serial.begin(115200); que en este caso es 115200 baudios. Si no se ven los mensajes, baja la velocidad tanto en el código, como en el monitor.

  • Ve como configurar y ejecutar este ejemplo.

#include <ESP8266WiFi.h>
#include <ESP8266WiFiMulti.h>
#include <ArduinoWebsockets.h>

const char* const SSID =
 "*******";
const char* const PASS =
 "*******";
const char* const HUELLA_DIGITAL =
 "*************";
const char* const URL =
 "wss://iot.gilbertopachec3.repl.co/";

using namespace websockets;
ESP8266WiFiMulti WiFiMulti;
WebsocketsClient  client;

const int BOTON = 0;
bool presionado = false;
bool conectado = false;
String valor;

void onMessageCallback(
  WebsocketsMessage message) {
 Serial.print("Recibe ");
 valor = message.data();
 digitalWrite(LED_BUILTIN,
  valor == "1" ? LOW : HIGH);
 Serial.println(message.data());
}

void onEventsCallback(
  WebsocketsEvent event,
  String data) {
 if (event ==
   WebsocketsEvent::ConnectionOpened) {
  Serial.println("Conectado.");
  conectado = true;
 } else if (event ==
   WebsocketsEvent::ConnectionClosed) {
  Serial.println("Desconectado.");
  conectado = false;
 }
}

void setup() {
 Serial.begin(115200);
 // Serial.setDebugOutput(true);
 Serial.println();
 Serial.println();
 Serial.println();
 pinMode(LED_BUILTIN, OUTPUT);
 pinMode(BOTON, INPUT);
 for (uint8_t t = 4; t > 0;
  t--) {
  Serial.printf(
   "ESPERANDO %d...\n", t);
  Serial.flush();
  delay(1000);
 }
 Serial.println(
  "Conectando WiFi...");
 WiFi.mode(WIFI_STA);
 WiFiMulti.addAP(SSID, PASS);
 while (WiFiMulti.run() !=
   WL_CONNECTED) {
  delay(500);
 }
 Serial.println("WiFi conectado.");
 client.
   onMessage(onMessageCallback);
 client.onEvent(onEventsCallback);
 client.
   setFingerprint(HUELLA_DIGITAL);
 Serial.println("Conectando WS.");
 client.connect(URL);
}

void loop() {
 client.poll();
 if (conectado) {
  bool actual = digitalRead(BOTON);
  if (!presionado &&
    actual == LOW) {
   client.send(valor);
   Serial.print("Envía");
   Serial.println(valor);
  }
  presionado = (actual == LOW);
 }
}
