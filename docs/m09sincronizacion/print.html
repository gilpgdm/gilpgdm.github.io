Versión para imprimir.

  • En esta lección se muestra un ejemplo de sincronización de bases de datos.

  • Puedes probar la app desde varios navegadores y dispositivos en https://sincronizacion.gilbertopachec2.repl.co/

    Firebase

    • https://pouchdb.com/

2. js / dao / conn.js

export const PASATIEMPO =
 "Pasatiempo"
const BD_NOMBRE = "sincronizacion"
const BD_VERSION = 1

/** @type {Promise<IDBDatabase>} */
export const conn =
 new Promise((resolve, reject) => {
  const solicitud = indexedDB.open(
   BD_NOMBRE, BD_VERSION)
  solicitud.onerror =
   () => reject(solicitud.error)
  solicitud.onsuccess =
   () => resolve(solicitud.result)
  /* Automáticamente se inicia una
   * transacción para cambio de
   * versión. */
  solicitud.onupgradeneeded =
   () => {
    const bd = solicitud.result
    /* Como hay cambio de versión,
     * borra el almacén si es que
     * existe. */
    if (bd.objectStoreNames
     .contains(PASATIEMPO)) {
     bd.deleteObjectStore(
      PASATIEMPO)
    }
    /* Crea el almacén "Pasatiempo"
     * con el campo llave "uuid".*/
    bd.
     createObjectStore(PASATIEMPO,
      { keyPath: "uuid" })
   }
 })

namespace srv\dao;

require_once "srv/dao/bdCrea.php";

use \PDO;

class AccesoBd
{

 private static ?PDO $con = null;

 static function getCon(): PDO
 {
  if (self::$con === null) {
   self::$con = self::conecta();
   self::prepara(self::$con);
  }
  return self::$con;
 }

 private static
 function conecta(): PDO
 {
  return new PDO(
   // cadena de conexión
   "sqlite:sincronizacion.db",
   // usuario
   null,
   // contraseña
   null,
   [PDO::ATTR_ERRMODE =>
   PDO::ERRMODE_EXCEPTION]
  );
 }

 private static
 function prepara(PDO $con)
 {
  bdCrea($con);
 }
}

import {
 bdConsulta
} from "../../lib/js/bdConsulta.js"
import {
 conn,
 PASATIEMPO
} from "./conn.js"

/**
 * Lista todos los objetos que no
 * tienen borrado lógico.
 * @returns {Promise<
 *  {
 *   uuid?:string,
 *   nombre: string
 *   modificacion?: number
 *   eliminado?: number} []>
 *  }
 */
export async function
 pasatiempoConsultaNoEliminados() {
 return bdConsulta(conn,
  [PASATIEMPO], (tx, resolve) => {
   const modelos = []
   const proc = tx.
    objectStore(PASATIEMPO).
    openCursor()
   proc.onsuccess = () => {
    const cursor = proc.result
    if (cursor === null) {
     resolve(modelos)
    } else {
     let modelo = cursor.value
     if (modelo.eliminado == 0) {
      modelos.push(modelo);
     }
     cursor.continue()
    }
   }
  })
}

5. srv / SrvSincro.php

<?php

require_once __DIR__ .
 "/../lib/php/autoload.php";
require_once
 "srv/dao/pasatiempoAgrega.php";
require_once
 "srv/dao/pasatiempoBusca.php";
require_once "srv/dao/" .
 "pasatiempoConsultaNoEliminados.php";
require_once
 "srv/dao/pasatiempoModifica.php";
require_once
 "lib/php/leeJson.php";
require_once
 "lib/php/copiaCampos.php";

use lib\php\Servicio;
use srv\modelo\Pasatiempo;

class SrvSincro extends Servicio
{
 protected
 function implementacion()
 {
  $lista = leeJson();
  if (!is_array($lista)) {
   $lista = [];
  }
  foreach ($lista as $json) {
   $modCli =
    $this->creaPasatiempo($json);
   $modServ =
    pasatiempoBusca($modCli->uuid);
   if ($modServ === false) {
    /* CONFLICTO. El objeto no ha
     * estado en el servidor.
     * AGREGARLO solamente si no
     * está eliminado. */
    if (!$modCli->eliminado) {
     pasatiempoAgrega($modCli);
    }
   } elseif (
    !$modServ->eliminado
    && $modCli->eliminado
   ) {
    /* CONFLICTO. El registro está
      * en el servidor, donde no se
      * ha eliminado, pero ha sido
      * eliminado en el cliente.
      * Gana el cliente, porque
      * optamos por no revivir lo
      * que se ha borrado. */
    pasatiempoModifica($modCli);
   } else if (
    !$modCli->eliminado
    && !$modServ->eliminado
   ) {
    /* CONFLICTO. El registro está
      * tanto en el servidor como
      * en el cliente. Los datos
      * pueden ser diferentes.
      * PREVALECE LA FECHA MÁS
      * GRANDE. Cuando gana el
      * servidor no se hace nada.*/
    if (
     $modCli->modificacion
     > $modServ->modificacion
    ) {
     /* La versión del cliente es
       * más nueva y prevalece. */
     pasatiempoModifica($modCli);
    }
   }
  }
  $lista =
   pasatiempoConsultaNoEliminados();
  return $lista;
 }

 function creaPasatiempo($json)
 {
  $pasatiempo = new Pasatiempo();
  copiaCampos($pasatiempo, $json);
  return $pasatiempo;
 }
}

$servicio = new SrvSincro();
$servicio->ejecuta();

}

 evt) {
 // Busca el contenido del cache.
 const cache = await abreElCache()
 const request = evt.request
 const response =
  await buscaLaPeticionEnElCache(
   cache, request)
 if (response === undefined) {
  /* Si no lo encuentra, lo
   * empieza a descargar de la red
   * y devuelve la promesa. */
  return fetch(request)
 } else {
  /* Si lo encuentra, devuelve la
   * respuesta que encontró en el
   * cache. */
  return response
 }
}

/**
 * @param {Promise<IDBDatabase>} bd
 * @param {string[]} entidades
 * @param { (
 *  tx:
 *   IDBTransaction,
 *  resolve:
 *   (value: any)=>void
 *  ) => void } funcion
 */
export async function bdConsulta(
 bd, entidades, funcion) {
 const con = await bd
 return new Promise(
  (resolve, reject) => {
   /* Inicia una transacción de
    * solo lectura. */
   const tx = con.transaction(
    entidades, "readonly")
   /* Al terminar con error ejecuta
    * la función reject. */
   tx.onerror =
    () => reject(tx.error)
   /* Estas son las operaciones de
    * la transacción. */
   funcion(tx, resolve)
  })
}

2. lib / js / bdEjecuta.js

/**
 * @param {Promise<IDBDatabase>} bd
 * @param {string[]} entidades
 * @param {(
 *   tx: IDBTransaction) => void
 *  } funcion
 */
export async function bdEjecuta(bd,
 entidades, funcion) {
 const con = await bd
 return new Promise(
  (resolve, reject) => {
   /* Inicia una transacción de
    * lectura y escritura. */
   const tx = con.transaction(
    entidades, "readwrite")
   /* Al terminar con éxito ejecuta
    * la función resolve. */
   tx.oncomplete = resolve
   /* Al terminar con error ejecuta
    * la función reject. */
   tx.onerror =
    () => reject(tx.error)
   /* Estas son las operaciones de
    * la transacción. */
   funcion(tx)
  })
}

4. lib / js / ejecuta.js

/**
 * Espera a que la promesa de un
 * fetch termine. Si hay error
 * lanza una excepción. Si no hay
 * error, interpreta la respuesta
 * del servidor como JSON y
 * la convierte en una literal de
 * objeto.
 * @param {Promise<Response>} fetch
 * @returns {Promise<any>}
 */
export
 async function ejecuta(fetch) {
 const resp = await fetch
 if (resp.ok) {
  return await resp.json()
 } else if (resp.status === 500) {
  throw new Error(
   await resp.text())
 } else {
  throw new Error(resp.statusText)
 }
}

8. lib / js / htmlentities.js

/**
 * Codifica un texto para que
 * cambie los caracteres
 * especiales y no se pueda
 * interpretar como HTML. Esta
 * técnica evita la inyección de
 * código.
 * @param {string} texto
 * @returns {string} un texto que
 *  no puede interpretarse como
 *  HTML.
 */
export function htmlentities(
 texto) {
 return texto.
  replace(/[<>"']/g, reemplaza)
}

/**
 * Devuelve una cadena de reemplazo
 * para evitar que el texto se
 * interprete como HTML.
 * @param {string} texto
 */
function reemplaza(texto) {
 switch (texto) {
  case "<": return "&lt;"
  case ">": return "&gt;"
  case '"': return "&quot;"
  case "'": return "&#039;"
  default: return texto
 }
}

6. lib / php / servicio.php

<?php

/**
 * Declara el espacio de nombre de
 * la clase. Es como la carpeta,
 * pero desde el punto de vista del
 * código.
 */
namespace lib\php;

/**
 * Clase base para implementar
 * servicios. Las clases derivadas
 * deben extender esta clase y
 * proporcionar el código para la
 * función llamada implementacion.
 */
abstract class Servicio
{

 abstract protected
 function implementacion();

 public function ejecuta()
 {
  try {
   $this->usaUtf8();
   $resultado =
    $this->implementacion();
   $this->devuelveJson($resultado);
  } catch (\Throwable $t) {
   $this->devuelveError($t);
  }
 }

 private function usaUtf8()
 {
  mb_internal_encoding("UTF-8");
 }

 private function devuelveJson(
  $resultado
 ) {
  header("Content-Type: "
   . "application/json");
  echo json_encode($resultado);
 }

 private function devuelveError(
  \Throwable $t
 ) {
  header(
   "Content-Type: text/plain"
  );
  http_response_code(500);
  echo $t->getMessage();
 }
}
