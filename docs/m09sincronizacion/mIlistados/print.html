  const
   FRECUENCIA_DE_SINCRONIZACION =
    20000

  async function preparaVista() {

    const render =
     await consultaLocal()

    sincroniza()
   } catch (e) {

    generaHtmlDeLista(lista)

  /** @param {Array} lista */
  function generaHtmlDeLista(
   lista) {
   let render = ""
   for (const modelo of lista) {
    const nombre =
     htmlentities(modelo.nombre)
    const searchParams =
     new URLSearchParams(
      [["uuid", modelo.uuid]])
    const params = htmlentities(
     searchParams.toString())
    render += /* HTML */
     `<li>
       <p>
 <a href="modifica.html?${params}">
  ${nombre}</a>
       </p>
      </li>`
   }
   return render
  }

  async function sincroniza() {
   try {
    if (navigator.onLine) {
     const todos = await
      pasatiempoConsultaTodos()
     const json = await enviaJson(
      todos, MetodoHttp.POST,
      "srv/SrvSincro.php")
     const listaSincronizada =
      Array.isArray(json)
       ? json
       : []
     await reemplazaObjetos(
      listaSincronizada)
     const render =
      generaHtmlDeLista(
       listaSincronizada)
     muestraRender(lista, render)
    }
   } catch (e) {
    muestraError(e)
   } finally {
    /* Espera un poco y lanza otra
     * sincronización. */
    setTimeout(sincroniza,
     FRECUENCIA_DE_SINCRONIZACION)
   }

2. js / dao / pasatiempoConsultaNoEliminados.js

import {
 bdConsulta
} from "../../lib/js/bdConsulta.js"
import {
 conn,
 PASATIEMPO
} from "./conn.js"

/**
 * Lista todos los objetos que no
 * tienen borrado lógico.
 * @returns {Promise<
 *  {
 *   uuid?:string,
 *   nombre: string
 *   modificacion?: number
 *   eliminado?: number} []>
 *  }
 */
export async function
 pasatiempoConsultaNoEliminados() {
 return bdConsulta(conn,
  [PASATIEMPO], (tx, resolve) => {
   const modelos = []
   const proc = tx.
    objectStore(PASATIEMPO).
    openCursor()
   proc.onsuccess = () => {
    const cursor = proc.result
    if (cursor === null) {
     resolve(modelos)
    } else {
     let modelo = cursor.value
     if (modelo.eliminado == 0) {
      modelos.push(modelo);
     }
     cursor.continue()
    }
   }
  })
}

4. js / dao / reemplazaObjetos.js

import {
 bdEjecuta
} from "../../lib/js/bdEjecuta.js"
import {
 conn,
 PASATIEMPO
} from "./conn.js"

/**
 * Borra el contenido del almacén y
 * guarda el contenido del listado.
 * @param {Array} lista nuevos
 *   datos. */
export async function
 reemplazaObjetos(lista) {
 return bdEjecuta(conn,
  [PASATIEMPO], tx => {
   const store =
    tx.objectStore(PASATIEMPO)
   store.clear()
   for (const modelo of lista) {
    store.add(modelo)
   }
  })
}

5. srv / SrvSincro.php

<?php

require_once __DIR__ .
 "/../lib/php/autoload.php";
require_once
 "srv/dao/pasatiempoAgrega.php";
require_once
 "srv/dao/pasatiempoBusca.php";
require_once "srv/dao/" .
 "pasatiempoConsultaNoEliminados.php";
require_once
 "srv/dao/pasatiempoModifica.php";
require_once
 "lib/php/leeJson.php";
require_once
 "lib/php/copiaCampos.php";

use lib\php\Servicio;
use srv\modelo\Pasatiempo;

class SrvSincro extends Servicio
{
 protected
 function implementacion()
 {
  $lista = leeJson();
  if (!is_array($lista)) {
   $lista = [];
  }
  foreach ($lista as $json) {
   $modCli =
    $this->creaPasatiempo($json);
   $modServ =
    pasatiempoBusca($modCli->uuid);
   if ($modServ === false) {
    /* CONFLICTO. El objeto no ha
     * estado en el servidor.
     * AGREGARLO solamente si no
     * está eliminado. */
    if (!$modCli->eliminado) {
     pasatiempoAgrega($modCli);
    }
   } elseif (
    !$modServ->eliminado
    && $modCli->eliminado
   ) {
    /* CONFLICTO. El registro está
      * en el servidor, donde no se
      * ha eliminado, pero ha sido
      * eliminado en el cliente.
      * Gana el cliente, porque
      * optamos por no revivir lo
      * que se ha borrado. */
    pasatiempoModifica($modCli);
   } else if (
    !$modCli->eliminado
    && !$modServ->eliminado
   ) {
    /* CONFLICTO. El registro está
      * tanto en el servidor como
      * en el cliente. Los datos
      * pueden ser diferentes.
      * PREVALECE LA FECHA MÁS
      * GRANDE. Cuando gana el
      * servidor no se hace nada.*/
    if (
     $modCli->modificacion
     > $modServ->modificacion
    ) {
     /* La versión del cliente es
       * más nueva y prevalece. */
     pasatiempoModifica($modCli);
    }
   }
  }
  $lista =
   pasatiempoConsultaNoEliminados();
  return $lista;
 }

 function creaPasatiempo($json)
 {
  $pasatiempo = new Pasatiempo();
  copiaCampos($pasatiempo, $json);
  return $pasatiempo;
 }
}

$servicio = new SrvSincro();
$servicio->ejecuta();

7. srv / dao / pasatiempoBusca.php

<?php

use srv\dao\AccesoBd;
use srv\modelo\Pasatiempo;

function pasatiempoBusca(
 string $uuid
): false|Pasatiempo {
 $con = AccesoBd::getCon();
 $stmt = $con->prepare(
  "SELECT
  PAS_UUID AS uuid,
  PAS_NOMBRE AS nombre,
  PAS_MODIFICACION AS modificacion,
  PAS_ELIMINADO AS eliminado
 FROM PASATIEMPO
 WHERE PAS_UUID = :uuid"
 );
 $stmt->execute(
  [":uuid" => $uuid]
 );
 $stmt->setFetchMode(
  PDO::FETCH_CLASS,
  Pasatiempo::class
 );
 return $stmt->fetch();
}
