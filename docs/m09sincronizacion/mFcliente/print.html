import {
 bdConsulta
} from "../../lib/js/bdConsulta.js"
import {
 conn,
 PASATIEMPO
} from "./conn.js"

/**
 * Lista todos los objetos que no
 * tienen borrado l칩gico.
 * @returns {Promise<
 *  {
 *   uuid?:string,
 *   nombre: string
 *   modificacion?: number
 *   eliminado?: number} []>
 *  }
 */
export async function
 pasatiempoConsultaNoEliminados() {
 return bdConsulta(conn,
  [PASATIEMPO], (tx, resolve) => {
   const modelos = []
   const proc = tx.
    objectStore(PASATIEMPO).
    openCursor()
   proc.onsuccess = () => {
    const cursor = proc.result
    if (cursor === null) {
     resolve(modelos)
    } else {
     let modelo = cursor.value
     if (modelo.eliminado == 0) {
      modelos.push(modelo);
     }
     cursor.continue()
    }
   }
  })
}
