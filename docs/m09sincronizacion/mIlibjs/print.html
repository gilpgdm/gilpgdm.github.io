1. lib / js / bdConsulta.js

/**
 * @param {Promise<IDBDatabase>} bd
 * @param {string[]} entidades
 * @param { (
 *  tx:
 *   IDBTransaction,
 *  resolve:
 *   (value: any)=>void
 *  ) => void } funcion
 */
export async function bdConsulta(
 bd, entidades, funcion) {
 const con = await bd
 return new Promise(
  (resolve, reject) => {
   /* Inicia una transacción de
    * solo lectura. */
   const tx = con.transaction(
    entidades, "readonly")
   /* Al terminar con error ejecuta
    * la función reject. */
   tx.onerror =
    () => reject(tx.error)
   /* Estas son las operaciones de
    * la transacción. */
   funcion(tx, resolve)
  })
}

2. lib / js / bdEjecuta.js

/**
 * @param {Promise<IDBDatabase>} bd
 * @param {string[]} entidades
 * @param {(
 *   tx: IDBTransaction) => void
 *  } funcion
 */
export async function bdEjecuta(bd,
 entidades, funcion) {
 const con = await bd
 return new Promise(
  (resolve, reject) => {
   /* Inicia una transacción de
    * lectura y escritura. */
   const tx = con.transaction(
    entidades, "readwrite")
   /* Al terminar con éxito ejecuta
    * la función resolve. */
   tx.oncomplete = resolve
   /* Al terminar con error ejecuta
    * la función reject. */
   tx.onerror =
    () => reject(tx.error)
   /* Estas son las operaciones de
    * la transacción. */
   funcion(tx)
  })
}

4. lib / js / ejecuta.js

/**
 * Espera a que la promesa de un
 * fetch termine. Si hay error
 * lanza una excepción. Si no hay
 * error, interpreta la respuesta
 * del servidor como JSON y
 * la convierte en una literal de
 * objeto.
 * @param {Promise<Response>} fetch
 * @returns {Promise<any>}
 */
export
 async function ejecuta(fetch) {
 const resp = await fetch
 if (resp.ok) {
  return await resp.json()
 } else if (resp.status === 500) {
  throw new Error(
   await resp.text())
 } else {
  throw new Error(resp.statusText)
 }
}

} from "./ejecuta.js"

 MetodoHttp
} from "./MetodoHttp.js"

/**
* @param {Object} datos
* @param {MetodoHttp} metodoHttp
* @param {string} url
*/
export async function enviaJson
 (datos, metodoHttp, url) {
 return await ejecuta(fetch(url
  , {
   method: metodoHttp
   , headers: {
    "Content-Type":
     "application/json"
   }
   , body: JSON.stringify(datos)
  }))
}

8. lib / js / htmlentities.js

/**
 * Codifica un texto para que
 * cambie los caracteres
 * especiales y no se pueda
 * interpretar como HTML. Esta
 * técnica evita la inyección de
 * código.
 * @param {string} texto
 * @returns {string} un texto que
 *  no puede interpretarse como
 *  HTML.
 */
export function htmlentities(
 texto) {
 return texto.
  replace(/[<>"']/g, reemplaza)
}

/**
 * Devuelve una cadena de reemplazo
 * para evitar que el texto se
 * interprete como HTML.
 * @param {string} texto
 */
function reemplaza(texto) {
 switch (texto) {
  case "<": return "&lt;"
  case ">": return "&gt;"
  case '"': return "&quot;"
  case "'": return "&#039;"
  default: return texto
 }
}
