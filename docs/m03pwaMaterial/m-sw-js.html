134 self.addEventListener("install",

136 console.log("El service worker se está instalando.")
137 evt.waitUntil(llenaElCache())

143 evt.respondWith(buscaLaRespuestaEnElCache(evt))
144 }

152async function llenaElCache() {
153 console.log("Intentando cargar caché:", CACHE, ".")
154 // Borra todos los cachés.
155 const keys = await caches.keys()
156 for (const key of keys) {
157 await caches.delete(key)
158 }
159 // Abre el caché de este service worker.
160 const cache = await caches.open(CACHE)
161 // Carga el listado de ARCHIVOS.
162 await cache.addAll(ARCHIVOS)
163 console.log("Cache cargado:", CACHE, ".")
164 console.log("Versión:", VERSION, ".")
165}
166
167/** @param {FetchEvent} evt */
168async function buscaLaRespuestaEnElCache(evt) {
169 // Abre el caché.
170 const cache = await caches.open(CACHE)
171 const request = evt.request
172 /* Busca la respuesta a la solicitud en el contenido del caché, sin
173 * tomar en cuenta la parte después del símbolo "?" en la URL. */
174 const response = await cache.match(request, { ignoreSearch: true })
175 if (response === undefined) {
176 /* Si no la encuentra, empieza a descargar de la red y devuelve
177 * la promesa. */
178 return fetch(request)
179 } else {
180 // Si la encuentra, devuelve la respuesta encontrada en el caché.
181 return response
182 }
183}
