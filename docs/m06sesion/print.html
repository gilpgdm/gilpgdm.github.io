class AccesoBd
{
 private static $con;

 public static
 function getCon(): \PDO
 {
  if (!self::$con) {
   self::$con = self::conecta();
   self::prepara();
  }
  return self::$con;
 }
 public static
 function conecta(): \PDO
 {
  return new PDO(
   "sqlite:phpauth.db",
   null,
   null,
   [PDO::ATTR_ERRMODE =>
   PDO::ERRMODE_EXCEPTION]
  );
 }
 public static function prepara()
 {
  bdCrea(self::getCon());
  $usuario =
   usuarioBusca("pepito");
  if (!$usuario) {
   $usuario = new Usuario();
   $usuario->cue = "pepito";
   $usuario->match = "cuentos";
   usuarioAgrega($usuario);
  }
 }
}

J. srv/dao/usuarioBusca.php

<?php
require_once __DIR__ .
 "/../modelo/Usuario.php";
require_once __DIR__ .
 "/AccesoBd.php";

function usuarioBusca(string $cue)
{
 $con = AccesoBd::getCon();
 $stmt = $con->prepare(
  "SELECT
    USU_ID as id,
    USU_CUE as cue,
    USU_MATCH as match
   FROM USUARIO
   WHERE USU_CUE = :cue"
 );
 $stmt->execute(
  [":cue" => $cue]
 );
 $stmt->setFetchMode(
  PDO::FETCH_CLASS,
  "Usuario"
 );
 return $stmt->fetch();
}

R. js/sesion.js

import {
 ejecuta
} from "../lib/js/ejecuta.js"
import {
 muestraError
} from "../lib/js/muestraError.js"
import {
 submit
} from "../lib/js/submit.js"

/** @type {HTMLFormElement|null} */
const sesion = document["sesion"]
/** @type {HTMLFormElement|null} */
const login = document["login"]

let esHumano = false

if (sesion !== null) {
 sesion.terminar.addEventListener(
  "click", terminaSesión)
}
if (login !== null) {
 login.addEventListener("submit",
  iniciaSesión)
 login.humano.addEventListener(
  "input", calculaQueSeaHumano)
}

recuperaSesion();

function calculaQueSeaHumano() {
 if (login !== null) {
  esHumano =
   login.humano.valueAsNumber > 0
 }
}

async function recuperaSesion() {
 try {
  if (sesion !== null) {
   const json = await ejecuta(
    fetch("srv/SrvSesion.php"))
   if (json.cue === "") {
    muestraLogin()
   } else {
    muestraSesion(json.cue)
   }
  }
 } catch (e) {
  muestraError(e)
 }
}

/** @param {Event} evt */
async function iniciaSesión(evt) {
 try {
  evt.preventDefault();
  if (!esHumano) {
   throw new Error(
    "Comprueba que eres humano.")
  }
  if (login !== null
   && sesion !== null) {
   const json = await submit(evt,
    "srv/SrvLogin.php", login)
   muestraSesion(json.cue)
  }
 } catch (e) {
  muestraError(e)
 }
}

async function terminaSesión() {
 try {
  await ejecuta(
   fetch("srv/SrvLogout.php"))
  muestraLogin()
 } catch (e) {
  muestraError(e)
 }
}

async function muestraLogin() {
 if (sesion !== null) {
  sesion.hidden = true
  sesion.cue.value = ""
 }
 if (login !== null) {
  login.cue.value = ""
  login.match.value = ""
  login.humano.valueAsNumber = 0
  esHumano = false
  login.hidden = false
 }
}

/**
 * @param {string} usuario
 */
async function
 muestraSesion(usuario) {
 if (sesion !== null) {
  sesion.hidden = false
  sesion.cue.value = usuario
 }
 if (login !== null) {
  login.match.value = ""
  login.hidden = true
 }
}

U. lib/js/ejecuta.js

/**
 * Espera a que la promesa de un
 * fetch termine. Si hay error
 * lanza una excepción. Si no hay
 * error, interpreta la respuesta
 * del servidor como JSON y
 * la convierte en una literal de
 * objeto.
 * @param {Promise<Response>} fetch
 * @returns {Promise<any>}
 */
export
 async function ejecuta(fetch) {
 const resp = await fetch
 if (resp.ok) {
  return await resp.json()
 } else if (
  resp.status === 500) {
  throw new Error(
   await resp.text())
 } else {
  throw new Error(
   resp.statusText)
 }
}

<?php
/**
 * Recupera un valor enviado al
 * servicio por medio de GET, POST
 * o cookie.
 */
function leeValor(
  string $nombre
): string {
  $valor =
    isset($_REQUEST[$nombre])
    ? $_REQUEST[$nombre]
    : "";
  return $valor;
}

Y. lib/php/Servicio.php

<?php
abstract class Servicio
{
 abstract protected
 function implementacion();
 public function ejecuta()
 {
  mb_internal_encoding("UTF-8");
  try {
   $resultado =
    $this->implementacion();
   echo json_encode($resultado);
  } catch (\Throwable $t) {
   http_response_code(500);
   echo $t->getMessage();
  }
 }
}
