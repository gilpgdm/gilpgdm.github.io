2. index.html

<!DOCTYPE html>
<html lang="es">

<head>
 <meta charset="utf-8">
 <meta name="viewport"
   content="width=device-width">
 <title>IoT</title>
 <script type="module"
   src="./lib/js/custom/indicador-cargando.js">
   </script>
</head>

<body>

 <h1>IoT</h1>

 <p>
  <output id="salida">
   <indicador-cargando>
   </indicador-cargando>
  </output>
 </p>

 <p>
  <button id="btnEnviar"
    type="button">
   Enviar
  </button>
 </p>

 <script type="module">
  import { muestraError }
   from "./lib/js/muestraError.js"
  import { errorEnWebSocket }
   from "./lib/js/errorEnWebSocket.js"
  import { txtConectado }
   from "./lib/js/txt/txtConectado.js"
  import { txtConexionTerminada }
   from "./lib/js/txt/txtConexionTerminada.js"

  try {

   let valor = ""

   /* Cambia la url por la de tu
    * servicio. */
    const ws = new WebSocket(
    "wss://iotwss.gilbertopachec2.repl.co")
   ws.onopen = webSocketAbierto
   ws.onmessage = recibeMensaje
   ws.onerror = errorEnWebSocket
   ws.onclose = webSocketCerrado

   function webSocketAbierto() {
    console.log(txtConectado())
    btnEnviar.addEventListener(
     "click", enviaMensaje)
   }

   /** @param {MessageEvent} evt */
   function recibeMensaje(evt) {
    valor = evt.data
    salida.value =
     valor === "1" ? "🔴" : "⚪"
   }

   function enviaMensaje() {
    ws.send(valor)
   }

  } catch (error) {
   muestraError(e)
  }

  /** @param {Event} evt */
  function webSocketCerrado(evt) {
   console.log(evt)
   alert(txtConexionTerminada())
   btnEnviar.removeEventListener(
    "click", enviaMensaje)
  }

 </script>

</body>

</html>

 connectedCallback() {

 }
