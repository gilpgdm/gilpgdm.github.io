1. Instrucciones

  • Descarga el ejemplo.

  • Revisa el proyecto en Replit con la URl https://replit.com/@GilbertoPacheco/pwamd?v=1

  • Prueba e instala el ejemplo con la URl https://pwamd.gilbertopacheco.repl.co/

  1. Descarga los siguientes archivos. Vienen dentro del proyecto compactado y se pueden descargar de la página de cada archivo, pero si los quieres descargar de su sitio original, sigue estas instrucciones:

    • Descarga el font Roboto usando https://google-webfonts-helper.herokuapp.com/fonts. Usa la opción Modern Browsers.

    • Descarga los íconos de Material Design en el repositorio https://github.com/google/material-design-icons. Descarga los archivos MaterialIcons-Regular.ttf y MaterialIcons-Regular.ttf de la carpeta font.

    • En https://developers.google.com/fonts/docs/material_icons. te indica como crear el archivo de estilos.

  2. Crea los íconos del proyecto con https://www.photopea.com/.

  3. Crea los íconos enmascarables con https://maskable.app/ a partir del archivo «icono2048.png».

  4. Usa o crea una cuenta de Google.

  5. Crea una cuenta de Replit usando la cuenta de Google.

  6. Crea un proyecto HTML, CSS, JS en Replit y edita o sube los archivos de este proyecto.

  7. Depura el proyecto.

    Cada vez que publiques los cambios en los archivos, debes modificar el valor de VERSION en el archivo «sw.js».

    Recuerda usar la combinación de teclas Ctrl+Mayúsculas+r para forzar que se actualice el navegador en caso de que no se vean los cambios.

  8. Crea la página de spotlight del proyecto.

  9. Instala y usa tu PWA en Windows.

  10. Instala y usa tu PWA en Android.

  11. Instala y usa tu PWA en iOS (iPhone y iPad).

4. js/recomienda.js

import { msg } from "../lib/msg.js"
import "./config.js";
import {
 CmDin
} from "../lib/cm-din.js"

const REQUERIDO = "*Requerido"

const form =
 document.querySelector("form")

/**
 * @type {HTMLSelectElement|null}
 */
const inGenero = document.
 querySelector("#inGenero")

/**
 * @type {HTMLOutputElement|null}
 */
const msgGenero = document.
 querySelector("#msgGenero")

/**
 * @type {HTMLOutputElement|null}
 */
const outSalida = document.
 querySelector("#outSalida")

if (inGenero && form) {
 form.addEventListener(
  "submit", procesa)
 inGenero.addEventListener("input",
  copiaMensajes)
}

function copiaMensajes() {
 msg(inGenero, msgGenero,
  REQUERIDO)
}

/**
 * @param {SubmitEvent} evt
 */
function procesa(evt) {
 evt.preventDefault()
 if (inGenero !== null
  && outSalida !== null) {
  try {
   if (inGenero.validity.valid) {
    const genero = inGenero.value
    if (genero === "pop") {
     outSalida.value =
      "Escucha a Ariana Grande."
    } else if (genero === "reg") {
     outSalida.value =
      "Escucha a Bad Bunny."
    } else {
     outSalida.value =
      "De ese género no conozco."
    }
   } else {
    outSalida.value = ""
   }
  } catch (e) {
   console.error(e)
   outSalida.value = e.message
  }
  if (outSalida.parentElement
   instanceof CmDin) {
   outSalida.parentElement.
    analiza()
  }
 }
}

9. sw.js

/* Este archivo debe estar
 * colocado en la carpeta raíz del
 * sitio.
 *
 * Cualquier cambio en el
 * contenido de este archivo hace
 * que el service worker se
 * reinstale.
 *
 * Normalmente se cambia el número
 * de la versión cuando cambia el
 * contenido de los archivos. */
const VERSION = "1.00"

/** Nombre del archivo de cache. */
const CACHE = "Rcm"

/** Archivos requeridos para que
 * la aplicación funcione fuera de
 * línea.
 */
const ARCHIVOS = [
 "css/estilos.css",
 "img/icono2048.png",
 "img/maskable_icon_x48.png",
 "img/maskable_icon_x72.png",
 "img/maskable_icon_x96.png",
 "img/maskable_icon_x128.png",
 "img/maskable_icon_x192.png",
 "img/maskable_icon_x384.png",
 "img/maskable_icon_x512.png",
 "img/maskable_icon.png",
 "js/config.js",
 "js/mi-nav.js",
 "js/recomienda.js",
 "lib/css/botones.css",
 "lib/css/cm-din.css",
 "lib/css/cn-app-bar.css",
 "lib/css/material-icons.css",
 "lib/css/mi-nav.css",
 "lib/css/roboto.css",
 "lib/fonts/MaterialIcons-Regular.codepoints",
 "lib/fonts/MaterialIcons-Regular.ttf",
 "lib/fonts/roboto-v29-latin-regular.woff",
 "lib/fonts/roboto-v29-latin-regular.woff2",
 "lib/cm-din.js",
 "lib/cn-app-bar.js",
 "lib/detecta.js",
 "lib/msg.js",
 "lib/regSw.js",
 "ayuda.html",
 "favicon.ico",
 "index.html",
 "site.webmanifest",
 "/"
];

if (self instanceof
 ServiceWorkerGlobalScope) {
 // Evento al empezar a instalar
 self.addEventListener("install",
  instala)

 // Evento al solicitar a la red
 self.addEventListener("fetch",
  descargaDatos)

 // Evento cuando está activo.
 self.addEventListener("activate",
  activo)
}

function activo() {
 console.log(
  "Service Worker activo.")
}

/**
 * @param {ExtendableEvent} evt
 */
function instala(evt) {
 console.log(
  "Service Worker instalando.")
 evt.waitUntil(cargaCache());
}

/**
 * @param {FetchEvent} evt
 */
function descargaDatos(evt) {
 if (
  evt.request.method === "GET") {
  evt.respondWith(usaCache(evt))
 }
}

async function cargaCache() {
 console.log(
  "Intentando cargar cache:",
  CACHE)
 // Borra todos los chaches.
 const keys = await caches.keys()
 for (const key of keys) {
  await caches.delete(key)
 }
 // Carga el nuevo contenido.
 const cache =
  await caches.open(CACHE)
 await cache.addAll(ARCHIVOS)
 console.
  log("Cache cargado:", CACHE)
 console.log("Versión:", VERSION)
}

/** @param {FetchEvent} evt */
async function usaCache(evt) {
 // Busca el contenido del cache.
 const cache =
  await caches.open(CACHE)
 const response =
  await cache.match(evt.request,
   { ignoreSearch: true })
 if (response) {
  /* Si lo encuentra, devuelve el
   * archivo del cache. */
  return response
 } else {
  /* Si no lo encuentra, lo
   * empieza a descargar de la red
   * y devuelve la promesa. */
  return fetch(evt.request)
 }
}

const CLASE_ELEVACION2 =
 "elevacion2"

export class CnAppBar
 extends HTMLElement {
 constructor() {
  super()
  /** @type {number} */
  this.posY = 0
  /** @type {boolean} */
  this.procesando = false
 }
 connectedCallback() {
  window.addEventListener(
   "scroll",
   () => this.onScroll())
 }
 onScroll() {
  this.posY = window.scrollY
  if (!this.procesando) {
   window.requestAnimationFrame(
    () => this.avanza())
  }
  this.procesando = true
 }
 avanza() {
  if (this.posY === 0) {
   this.classList.
    remove(CLASE_ELEVACION2)
  } else {
   this.classList.
    add(CLASE_ELEVACION2)
  }
  this.procesando = false
 }
}
customElements.
 define("cn-app-bar", CnAppBar)

37. lib/msg.js

/** Sin un elemento tiene un
 * mensaje de validación, lo
 * muestra en su elemento de ayuda
 * y si no, muestra un texto de
 * ayuda.
 * @param {HTMLInputElement|
 *  HTMLTextAreaElement|
 *  HTMLSelectElement|
 *  HTMLOutputElement|null} e
 * referencia a un elemento que
 * contiene datos.
 * @param {HTMLOutputElement|
 *  null} out
 * elemento de ayuda para e.
 * @param {string} ok mensaje de
 * ayuda cuando el estado de e es
 * válido. */
export function msg(e, out, ok) {
  if (e && out) {
    if (e.validity.valid) {
      out.value = ok
    } else {
      out.value =
        e.validationMessage
    }
  }
}
