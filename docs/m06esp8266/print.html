B. Blink.ino

Ve como capturar y ejecutar este ejemplo.

Ve como desplegar este ejemplo en Tinkercad y compartirlo.

/* Este programa es un derivado de
   ESP8266 Blink by Simon Peter */

/** Función que se invoca una sola
    vez al inicio del programa. */
void setup() {
 /* Todas las placas compatibles
    con Arduino incluyen un led
    de prueba ligado a un pin
    cuyo número está definido por
    la constante LED_BUILTIN.
    Inicializa el pin para
    LED_BUILTIN como salida. */
 pinMode(LED_BUILTIN, OUTPUT);
}

/* Función que se invoca
   repetidamente mientras el
   programa esté activo. */
void loop() {
 /* Enciende el LED_BUILTIN (HIGH
    es el nivel de voltaje). */
 digitalWrite(LED_BUILTIN, HIGH);
 delay(1000); // Espera 1 segundo
 /* Apaga el LED_BUILTIN haciendo
    que el nivel de voltaje sea
    LOW. */
 digitalWrite(LED_BUILTIN, LOW);
 delay(1000);
}

C. Button.ino

Ve como desplegar este ejemplo en Tinkercad y compartirlo.

/* Este es un derivado de Button
   por DojoDave y Tom Igoe */

/* Pin para el botón. */
const int BOTON = 0;

int estadoDelBoton = 0;

void setup() {
 /* Inicializa el pin para
    LED_BUILTIN como salida. */
 pinMode(LED_BUILTIN, OUTPUT);
 /* Initializa el pin para
    BOTON como entrada. */
 pinMode(BOTON, INPUT);
}

void loop() {
 estadoDelBoton =
   digitalRead(BOTON);
 /* Checa si el botón está
    presionado. Esto es, si
    estadoDelBotón es LOW. */
 if (estadoDelBoton == LOW) {
  // Enciende el LED_BUILTIN.
  digitalWrite(LED_BUILTIN, HIGH);
 } else {
  // Apaga el LED_BUILTIN.
  digitalWrite(LED_BUILTIN, LOW);
 }
}

D. Dispositivo.ino

  • En el siguiente código, sustituye la configuración por los datos de tu red WiFi.

  • La URL y el puerto deben coincidir con tu dervidor de MQTT. Si usas test.mosquitto.org, conserva los valores mostrados más adelante.

  • Para monitorear su funcionamiento, debes hacer clic en el botón de arriba a la derecha en la ventana de Arduino, que tiene una lupa y tiene el mensaje flotante que dice Monitor Serie. Debes ajustar los baudios a la misma velocidad que en el código, donde dice Serial.begin(115200); que en este caso es 115200 baudios. Si no se ven los mensajes, baja la velocidad tanto en el código, como en el monitor.

  • Ve como configurar y ejecutar este ejemplo. No hagas caso a la parte de los certificados,

#include "EspMQTTClient.h"

const char *const SSID =
    "Galaxy A723C85";
const char *const PASS =
    "bdoi1764";
const char *const URL =
    "test.mosquitto.org";
const uint16_t PUERTO =
    1883;
const char *const CLIENT_ID =
    "gilpgdmIoT-esp32-1";
const char *const TOPICO_FOCO =
    "gilpgdm/IoT/foco";

EspMQTTClient cliente(
    SSID,
    PASS,
    URL,
    0, // Usuario opcional
    0, // Contraseña opcional
    CLIENT_ID,
    PUERTO);
const int BOTON = 0;
bool presionado = false;
String valor;

void recibeMensaje(
    const String &payload)
{
 Serial.print("Recibe ");
 valor = payload;
 digitalWrite(LED_BUILTIN,
              valor == "1"
                  ? HIGH
                  : LOW);
 Serial.println(payload);
}

void enviaMensajeMqtt(
    String valorAEnviar,
    String topico)
{
 cliente.publish(topico,
                 valorAEnviar);
}

void onConnectionEstablished()
{
 Serial.println("Conectado.");
 cliente.subscribe(
     TOPICO_FOCO, recibeMensaje);
 enviaMensajeMqtt(
     valor, TOPICO_FOCO);
}

void setup()
{
 Serial.begin(115200);
 pinMode(BOTON, INPUT);
}

void loop()
{
 cliente.loop();
 if (cliente.isConnected())
 {
  bool actual =
      digitalRead(BOTON);
  if (!presionado &&
      actual == LOW)
  {
   enviaMensajeMqtt(
       valor == "1" ? "0" : "1",
       TOPICO_FOCO);
   Serial.print("Envía ");
   Serial.println(valor);
  }
  presionado = (actual == LOW);
 }
}
