C. Dispositivo.ino

  • En el siguiente código, sustituye la configuración por los datos de tu red WiFi. La URL y el certificado deben corresponder a tu proyecto de IoT.

  • Las siguientes instrucciones son para obtener el certificado:

    1. Usa Chrome para abrir la URL del código de la lección IoT. En este caso es https://iot2.gilbertopacheco.repl.co/.

    2. Haz clic en el candadito que está junto a la URL. Se abre un menú contextual.

    3. Haz clic en donde dice La conexión es segura. Aparece un cuadro de diálogo.

    4. Haz clic en donde dice El certificado es válido. Aparece un cuadro de diálogo.

    5. Haz clic en la pestaña que dice Ruta de certificación.

    6. Haz clic en la la raíz del árbol, que en este caso dice ISGR Root X1.

    7. Haz clic en el botón que dice Ver certificado.

    8. Observa las fechas de validez para que sepas cuando caduca. En ese caso tendrás que repetir este procedimiento.

    9. Selecciona la pestaña Detalles.

    10. Haz clic en copiar en archivo…. Aparece un asistente.

    11. Haz clic en Siguiente.

    12. Haz clic en el botón de radio que dice X.509 codificado base 64 (.CER).

    13. Haz clic en el botón que dice Siguiente.

    14. Haz clic en el botón que dice Examinar para que selecciones la ruta del archivo e introduzcas su nombre. En este ejemplo se selecciona la carpeta de descargas y se le pone por nombre isgr.cer

    15. Haz clic en el botón que dice Siguiente.

    16. Haz clic en el botón que dice Finalizar. Si todo sale bien aparece un cuadro de diálogo informando que la exportación se realizó correctamente.

    17. Haz clic en el botón que dice Aceptar.

    18. Copia línea por línea el contenido del archivo generado en las cadenas de la constante CERTIFICADO. Conserva las comillas y la secuencia \n antes de cerrar comillas.

  • Para monitorear su funcionamiento, debes hacer clic en el botón de arriba a la derecha en la ventana de Arduino, que tiene una lupa y tiene el mensaje flotante que dice Monitor Serie. Debes ajustar los baudios a la misma velocidad que en el código, donde dice Serial.begin(115200); que en este caso es 115200 baudios. Si no se ven los mensajes, baja la velocidad tanto en el código, como en el monitor.

void onMessageCallback(
  WebsocketsMessage message) {
 Serial.print("Recibe ");
 valor = message.data();
 digitalWrite(LED_BUILTIN,
  valor == "1" ? HIGH : LOW);
 Serial.println(message.data());
}

void onEventsCallback(
  WebsocketsEvent event,
  String data) {
 if (event == WebsocketsEvent::
   ConnectionOpened) {
  Serial.println("Conectado.");
  conectado = true;
 } else if (event ==
   WebsocketsEvent::
   ConnectionClosed) {
  Serial.println("Desconectado.");
  conectado = false;
 }
}

void setup() {
 Serial.begin(115200);
 pinMode(LED_BUILTIN, OUTPUT);
 pinMode(BOTON, INPUT);
 WiFi.begin(SSID, PASS);
 Serial.print("Conectando WiFi.");
 for (int i = 0; i < 10 &&
   WiFi.status() != WL_CONNECTED;
   i++) {
  Serial.print(".");
  delay(1000);
 }
 Serial.println();
 if (WiFi.status() !=
   WL_CONNECTED) {
  Serial.println(
    "WiFi desconectado.");
 } else {
  Serial.println(
    "WiFi conectado.");
  client.
   onMessage(onMessageCallback);
  client.onEvent(onEventsCallback);
  client.setCACert(CERTIFICADO);
  Serial.println("Conectando WS.");
  bool conectado =
    client.connect(URL);
  if (conectado) {
    Serial.println(
      "WS Conectado.");    
  } else {
    Serial.println(
      "WS NO Conectado.");    
  }
 }
}

void loop() {
 if (client.available()) {
  client.poll();
  if (conectado) {
   bool actual =
     digitalRead(BOTON);
   if (!presionado &&
     actual == LOW) {
    client.send(valor);
    Serial.print("Envía ");
    Serial.println(valor);
   }
   presionado = (actual == LOW);
  }
 }
}
