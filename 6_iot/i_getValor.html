const cliente =
 require("./cliente")

async function getValor() {
 const client = cliente()
 try {
  await client.connect()
  const iot = client.db("iot")
  const dispositivo =
   iot.collection("dispositivo")
  /** @type {import("./tipos").
   *       dispositivo|null} */
  const modelo = await dispositivo.
   findOne({ _id: "1" })
  console.log("Recibe", "bd",
   modelo)
  if (modelo !== null) {
   return modelo.valor;
  } else {
   return "0";
  }
 } catch (e) {
  console.log(e);
  return "0";
 } finally {
  client.close();
 }
}

module.exports = getValor;
